199bool vec3_equal(Vec3 a, Vec3 b) {
200 static float EPSILON = 1e-6f;
201
202 // Pack Vec3 into __m128 with padding (e.g. 0.0f)
203 __m128 va = _mm_set_ps(0.0f, a.z, a.y, a.x);
204 __m128 vb = _mm_set_ps(0.0f, b.z, b.y, b.x);
205
206 // Compute absolute difference
207 __m128 diff = _mm_sub_ps(va, vb);
208 __m128 abs_diff = _mm_andnot_ps(_mm_set1_ps(-0.0f), diff); // fabs
209
210 // Compare with EPSILON
211 __m128 eps = _mm_set1_ps(EPSILON);
212 __m128 cmp = _mm_cmplt_ps(abs_diff, eps); // abs_diff < EPSILON
213
214 // Extract bitmask
215 int mask = _mm_movemask_ps(cmp);
216
217 // Only check the first 3 lanes (bits 0–2)
218 return (mask & 0x7) == 0x7;
219}

221static inline bool vec4_equal(Vec4 a, Vec4 b) {
222 static float EPSILON = 1e-6f;
223
224 // Load vectors into SIMD registers
225 __m128 va = _mm_set_ps(a.w, a.z, a.y, a.x);
226 __m128 vb = _mm_set_ps(b.w, b.z, b.y, b.x);
227
228 // Compute absolute difference
229 __m128 diff = _mm_sub_ps(va, vb);
230 __m128 abs_diff = _mm_andnot_ps(_mm_set1_ps(-0.0f), diff); // clear sign bit (fabs)
231
232 // Compare with EPSILON
233 __m128 eps = _mm_set1_ps(EPSILON);
234 __m128 cmp = _mm_cmplt_ps(abs_diff, eps); // true if abs_diff < EPSILON
235
236 // All bits set (0xFFFFFFFF) in each lane means equal
237 return _mm_movemask_ps(cmp) == 0xF;
238}

385 // Rodrigues' rotation formula

387 float sin_angle = sinf(angle);
388 Vec3 norm_axis = vec3_normalize(axis);

390 Vec3 term1 = vec3_mul(v, cos_angle);
391 Vec3 term2 = vec3_mul(vec3_cross(norm_axis, v), sin_angle);

531 // Create SIMD vectors for cos and sin
532 __m128 cos_vec = _mm_set1_ps(cos_angle);
533 __m128 sin_vec = _mm_set1_ps(sin_angle);

535 // Extract components
536 __m128 x = _mm_shuffle_ps(v.m128, v.m128, _MM_SHUFFLE(0, 0, 0, 0));

542
543 // Combine results
544 SimdVec2 result;
545 result.m128 = _mm_unpacklo_ps(x_rot, y_rot);
546 return result;

597static inline float simd_vec2_dot(SimdVec2 a, SimdVec2 b) {
598 // Perform element-wise multiplication of the x and y components
599 __m128 mul = _mm_mul_ps(a.m128,
600 b.m128); // Multiply the entire 128-bit register (containing 4 floats)
601
602 // Store the result in a float array to sum up the first two components
603 float result[4];
604 _mm_storeu_ps(result, mul); // Store the result of the multiplication

607 return result[0] + result[1]; // Dot product is just x1 * x2 + y1 * y2

686static inline SimdVec3 simd_vec3_rotate(SimdVec3 v, SimdVec3 axis, float angle) {
687 // Compute cos and sin of angle
688 float cos_angle = cosf(angle);
689 float sin_angle = sinf(angle);
690
691 // Create SIMD vectors for cos and sin
692 __m256 cos_vec = _mm256_set1_ps(cos_angle);
693 __m256 sin_vec = _mm256_set1_ps(sin_angle);
694
695 // Normalize the axis vector
696 __m256 axis_len = _mm256_sqrt_ps(_mm256_dp_ps(axis.m256, axis.m256, 0xFF));
697 __m256 norm_axis = _mm256_div_ps(axis.m256, axis_len);
698
699 // Compute dot product (v · axis)
700 __m256 dot = _mm256_dp_ps(v.m256, norm_axis, 0xFF);
701
702 // Rodrigues' rotation formula:
703 // v_rot = v*cosθ + (axis × v)*sinθ + axis*(axis·v)(1 - cosθ)
704
705 // Term 1: v*cosθ
706 __m256 term1 = _mm256_mul_ps(v.m256, cos_vec);
707
708 // Term 2: (axis × v)*sinθ
709 __m256 cross = _mm256_sub_ps(_mm256_mul_ps(_mm256_shuffle_ps(norm_axis, norm_axis, _MM_SHUFFLE(3, 0, 2, 1)),
710 _mm256_shuffle_ps(v.m256, v.m256, _MM_SHUFFLE(3, 1, 0, 2))),
711 _mm256_mul_ps(_mm256_shuffle_ps(norm_axis, norm_axis, _MM_SHUFFLE(3, 1, 0, 2)),
712 _mm256_shuffle_ps(v.m256, v.m256, _MM_SHUFFLE(3, 0, 2, 1))));
713 __m256 term2 = _mm256_mul_ps(cross, sin_vec);
714
715 // Term 3: axis*(axis·v)(1 - cosθ)
716 __m256 one_minus_cos = _mm256_sub_ps(_mm256_set1_ps(1.0f), cos_vec);
717 __m256 term3 = _mm256_mul_ps(norm_axis, _mm256_mul_ps(dot, one_minus_cos));
718
719 // Combine all terms
720 SimdVec3 result;
721 result.m256 = _mm256_add_ps(_mm256_add_ps(term1, term2), term3);
722 return result;
723}

786static inline SimdVec4 simd_vec4_rotate_x(SimdVec4 v, float angle) {
787 float cos_angle = cosf(angle);
788 float sin_angle = sinf(angle);
789
790 __m128 cos_vec = _mm_set1_ps(cos_angle);
791 __m128 sin_vec = _mm_set1_ps(sin_angle);
792
793 // Extract y and z components
794 __m128 y = _mm_shuffle_ps(v.m128, v.m128, _MM_SHUFFLE(1, 1, 1, 1));
795 __m128 z = _mm_shuffle_ps(v.m128, v.m128, _MM_SHUFFLE(2, 2, 2, 2));
796
797 // Perform rotation: y' = y*cos - z*sin; z' = y*sin + z*cos
798 __m128 y_rot = _mm_sub_ps(_mm_mul_ps(y, cos_vec), _mm_mul_ps(z, sin_vec));
799 __m128 z_rot = _mm_add_ps(_mm_mul_ps(y, sin_vec), _mm_mul_ps(z, cos_vec));
800
801 // Create masks for y and z components
802 const __m128 y_mask = _mm_castsi128_ps(_mm_set_epi32(0, 0, ~0, 0));
803 const __m128 z_mask = _mm_castsi128_ps(_mm_set_epi32(0, ~0, 0, 0));
804
805 // Combine results
806 SimdVec4 result;
807 result.m128 = _mm_blendv_ps(v.m128, y_rot, y_mask);
808 result.m128 = _mm_blendv_ps(result.m128, z_rot, z_mask);
809 return result;

852 __m128 x = _mm_shuffle_ps(v.m128, v.m128, _MM_SHUFFLE(0, 0, 0, 0));
853 __m128 y = _mm_shuffle_ps(v.m128, v.m128, _MM_SHUFFLE(1, 1, 1, 1));
854
855 // Perform rotation: x' = x*cos - y*sin; y' = x*sin + y*cos
856 __m128 x_rot = _mm_sub_ps(_mm_mul_ps(x, cos_vec), _mm_mul_ps(y, sin_vec));
857 __m128 y_rot = _mm_add_ps(_mm_mul_ps(x, sin_vec), _mm_mul_ps(y, cos_vec));
858
859 // Create masks for x and y components
860 const __m128 x_mask = _mm_castsi128_ps(_mm_set_epi32(0, 0, 0, ~0));
861 const __m128 y_mask = _mm_castsi128_ps(_mm_set_epi32(0, 0, ~0, 0));
862
863 // Combine results
864 SimdVec4 result;
865 result.m128 = _mm_blendv_ps(v.m128, x_rot, x_mask);
866 result.m128 = _mm_blendv_ps(result.m128, y_rot, y_mask);
867 return result;
868}
