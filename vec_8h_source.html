384static inline SimdVec2 vec2_rotate(SimdVec2 v, float angle) {
385 float c = cosf(angle);
386 float s = sinf(angle);
387
388 // Broadcast components to all lanes for vectorized linear combination
389 simd_vec_t x = simd_splat_x(v.v);
390 simd_vec_t y = simd_splat_y(v.v);
391
392 // Rotation matrix columns:
393 // col0 = {c, s, 0, 0} - maps X component
394 // col1 = {-s, c, 0, 0} - maps Y component
395 simd_vec_t c0 = simd_set(c, s, 0.0f, 0.0f);
396 simd_vec_t c1 = simd_set(-s, c, 0.0f, 0.0f);
397
398 // result = x * col0 + y * col1
399 return (SimdVec2){.v = simd_add(simd_mul(x, c0), simd_mul(y, c1))};
400}

758static inline SimdVec3 vec3_perpendicular(SimdVec3 v) {
759 // Strategy: Cross v with the world axis it is LEAST parallel to.
760 // If |v.y| < 0.9, cross with Y-axis (0,1,0).
761 // Otherwise cross with X-axis (1,0,0).
762
763 // We access scalars here because conditional logic is cleaner in scalar
764 // than trying to construct a branchless SIMD mask for this specific logic.
765 SimdVec3 axis;
766 if (fabsf(v.y) < 0.99f) {
767 axis = vec3_load((Vec3){0.0f, 1.0f, 0.0f}); // Up
768 } else {
769 axis = vec3_load((Vec3){1.0f, 0.0f, 0.0f}); // Right
770 }
771
772 return vec3_normalize(vec3_cross(v, axis));
773}

952static inline SimdVec4 vec4_rotate_x(SimdVec4 v, float angle) {
953 float c = cosf(angle);
954 float s = sinf(angle);
955
956 // Apply rotation matrix to Y and Z components
957 // X and W are preserved
958 float ny = v.y * c - v.z * s;
