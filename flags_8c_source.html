561static FlagStatus parse_value(Flag* flag, const char* str) {
562 if (!str) return FLAG_ERROR_MISSING_VALUE;
563 char* endptr = NULL;
564 errno = 0;
565
566 switch (flag->type) {
567 case TYPE_BOOL: {
568 bool val = true;
569 if (strcasecmp(str, "false") == 0 || strcmp(str, "0") == 0) val = false;
570 *(bool*)flag->value_ptr = val;
571 break;
572 }
573 case TYPE_CHAR:
574 if (!str || strlen(str) == 0) return FLAG_ERROR_MISSING_VALUE;
575 if (strlen(str) != 1) return FLAG_ERROR_INVALID_ARGUMENT;
576 *(char*)flag->value_ptr = str[0];
577 break;
578 case TYPE_STRING:
579 *(char**)flag->value_ptr = (char*)str;
580 break;
581
582 // Signed Integers
583 case TYPE_INT8:
584 case TYPE_INT16:
585 case TYPE_INT32:
586 case TYPE_INT64: {
587 long long val = strtoll(str, &endptr, 10);
588 if (endptr == str || *endptr != '\0' || errno == ERANGE) return FLAG_ERROR_INVALID_NUMBER;
589
590 if (flag->type == TYPE_INT8 && !check_range_int(val, INT8_MIN, INT8_MAX)) return FLAG_ERROR_INVALID_NUMBER;
591 if (flag->type == TYPE_INT16 && !check_range_int(val, INT16_MIN, INT16_MAX))
592 return FLAG_ERROR_INVALID_NUMBER;
593 if (flag->type == TYPE_INT32 && !check_range_int(val, INT32_MIN, INT32_MAX))
594 return FLAG_ERROR_INVALID_NUMBER;
595
596 // Assign based on type size
597 if (flag->type == TYPE_INT8)
598 *(int8_t*)flag->value_ptr = (int8_t)val;
599 else if (flag->type == TYPE_INT16)
600 *(int16_t*)flag->value_ptr = (int16_t)val;
601 else if (flag->type == TYPE_INT32)
602 *(int32_t*)flag->value_ptr = (int32_t)val;
603 else
604 *(int64_t*)flag->value_ptr = (int64_t)val;
605 break;
606 }
607
608 // Unsigned Integers
609 case TYPE_UINT8:
610 case TYPE_UINT16:
611 case TYPE_UINT32:
612 case TYPE_UINT64:
613 case TYPE_SIZE_T: {
614 if (str[0] == '-') return FLAG_ERROR_INVALID_NUMBER;
615 unsigned long long val = strtoull(str, &endptr, 10);
616 if (endptr == str || *endptr != '\0' || errno == ERANGE) return FLAG_ERROR_INVALID_NUMBER;
617
618 if (flag->type == TYPE_UINT8 && !check_range_uint(val, UINT8_MAX)) return FLAG_ERROR_INVALID_NUMBER;
619 if (flag->type == TYPE_UINT16 && !check_range_uint(val, UINT16_MAX)) return FLAG_ERROR_INVALID_NUMBER;
620 if (flag->type == TYPE_UINT32 && !check_range_uint(val, UINT32_MAX)) return FLAG_ERROR_INVALID_NUMBER;
621 if (flag->type == TYPE_SIZE_T && !check_range_uint(val, SIZE_MAX)) return FLAG_ERROR_INVALID_NUMBER;
622
623 if (flag->type == TYPE_UINT8)
624 *(uint8_t*)flag->value_ptr = (uint8_t)val;
625 else if (flag->type == TYPE_UINT16)
626 *(uint16_t*)flag->value_ptr = (uint16_t)val;
627 else if (flag->type == TYPE_UINT32)
628 *(uint32_t*)flag->value_ptr = (uint32_t)val;
629 else if (flag->type == TYPE_SIZE_T)
630 *(size_t*)flag->value_ptr = (size_t)val;
631 else
632 *(uint64_t*)flag->value_ptr = (uint64_t)val;
633 break;
634 }
635
636 case TYPE_FLOAT: {
637 float val = strtof(str, &endptr);
638 if (endptr == str || *endptr != '\0' || errno == ERANGE) return FLAG_ERROR_INVALID_NUMBER;
639 *(float*)flag->value_ptr = val;
640 break;
641 }
642 case TYPE_DOUBLE: {
643 double val = strtod(str, &endptr);
644 if (endptr == str || *endptr != '\0' || errno == ERANGE) return FLAG_ERROR_INVALID_NUMBER;
645 *(double*)flag->value_ptr = val;
646 break;
647 }
648 default:
649 return FLAG_ERROR_INVALID_ARGUMENT;
650 }
651 return FLAG_OK;
652}

675FlagStatus flag_parse(FlagParser* fp, int argc, char** argv) {
676 if (!fp || argc < 1) return FLAG_ERROR_INVALID_ARGUMENT;
677
678 int i = 1;
679 bool end_of_opts = false;
680
681 while (i < argc) {
682 char* arg = argv[i];
683
684 // 1. End of Options
685 if (!end_of_opts && strcmp(arg, "--") == 0) {
686 end_of_opts = true;
687 i++;
688 continue;
689 }
690
691 // 2. Positionals (or Subcommands)
692 if (end_of_opts || arg[0] != '-') {
693 // Check subcommands
694 if (!end_of_opts && fp->cmd_count > 0) {
695 for (size_t c = 0; c < fp->cmd_count; c++) {
696 if (strcmp(fp->subcommands[c]->name, arg) == 0) {
697 fp->active_subcommand = fp->subcommands[c];
698 // Recursive call for subcommand
699 return flag_parse(fp->active_subcommand, argc - i, argv + i);
700 }
701 }
702 }
703
704 // Add positional
705 if (fp->pos_count >= fp->pos_capacity) {
706 size_t new_cap = (fp->pos_capacity == 0) ? INITIAL_CAPACITY : fp->pos_capacity * 2;
707 fp->positional_args = (char**)xrealloc(fp->positional_args, new_cap * sizeof(char*));
708 fp->pos_capacity = new_cap;
709 }
710 fp->positional_args[fp->pos_count++] = arg;
711 i++;
712 continue;
713 }
714
715 // 3. Long Flags
716 if (strncmp(arg, "--", 2) == 0) {
717 char* name_start = arg + 2;
718 if (strcmp(name_start, "help") == 0) {
719 flag_print_usage(fp);
720 exit(0);
721 }
722
723 char* eq = strchr(name_start, '=');
724 char name_buf[MAX_FLAG_NAME_LEN];
725 const char* val_str = NULL;
726 const char* lookup_name = name_start;
727
728 if (eq) {
729 size_t len = (size_t)(eq - name_start);
730 if (len >= MAX_FLAG_NAME_LEN) len = MAX_FLAG_NAME_LEN - 1;
731 strncpy(name_buf, name_start, len);
732 name_buf[len] = '\0';
733 lookup_name = name_buf;
734 val_str = eq + 1;
735 }
736
737 Flag* f = find_flag_long(fp, lookup_name);
738 if (!f) {
739 set_error(fp, "Unknown flag: --%s", lookup_name);
740 return FLAG_ERROR_UNKNOWN_FLAG;
741 }
742
743 f->is_present = true;
744
745 if (f->type == TYPE_BOOL) {
746 bool b = true;
747 if (val_str && (strcasecmp(val_str, "false") == 0 || strcmp(val_str, "0") == 0)) b = false;
748 *(bool*)f->value_ptr = b;
749 } else {
750 if (!val_str) {
751 if (i + 1 < argc && argv[i + 1][0] != '-')
752 val_str = argv[++i];
753 else {
754 set_error(fp, "Flag --%s requires a value", f->name);
755 return FLAG_ERROR_MISSING_VALUE;
756 }
757 }
758 FlagStatus s = parse_value(f, val_str);
759 if (s != FLAG_OK) {
760 set_error(fp, "Invalid value for --%s: '%s' (Type mismatch or overflow)", f->name, val_str);
761 return s;
762 }
763 }
764 i++;
765 continue;
766 }
767
768 // 4. Short Flags
769 if (arg[0] == '-') {
770 size_t len = strlen(arg);
771 for (size_t k = 1; k < len; k++) {
772 char c = arg[k];
773 Flag* f = find_flag_short(fp, c);
774 if (!f) {
775 set_error(fp, "Unknown short flag: -%c", c);
776 return FLAG_ERROR_UNKNOWN_FLAG;
777 }
778 f->is_present = true;
779
780 if (f->type == TYPE_BOOL) {
781 *(bool*)f->value_ptr = true;
782 } else {
783 const char* val_str = NULL;
784 if (k + 1 < len) {
785 val_str = &arg[k + 1];
786 if (*val_str == '=') val_str++;
787 FlagStatus s = parse_value(f, val_str);
788 if (s != FLAG_OK) {
789 set_error(fp, "Invalid value for -%c (Type mismatch or overflow)", c);
790 return s;
791 }
792 break; // consumed rest
793 } else {
794 if (i + 1 < argc && argv[i + 1][0] != '-') {
795 val_str = argv[++i];
796 FlagStatus s = parse_value(f, val_str);
797 if (s != FLAG_OK) {
798 set_error(fp, "Invalid value for -%c (Type mismatch or overflow)", c);
799 return s;
800 }
801 } else {
802 set_error(fp, "Flag -%c requires a value", c);
803 return FLAG_ERROR_MISSING_VALUE;
804 }
805 }
806 }
807 }
808 i++;
809 continue;
810 }
811 }
812
813 // Validation
814 for (size_t k = 0; k < fp->flag_count; k++) {
815 Flag* f = &fp->flags[k];
816 if (f->required && !f->is_present) {
817 set_error(fp, "Missing required flag: --%s", f->name);
818 return FLAG_ERROR_REQUIRED_MISSING;
819 }
820 if (f->is_present && f->validator) {
821 const char* err = NULL;
822 if (!f->validator(f->value_ptr, &err)) {
823 set_error(fp, "Validation failed for --%s: %s", f->name, err ? err : "invalid");
824 return FLAG_ERROR_VALIDATION;
825 }
826 }
827 }
828
829 return FLAG_OK;
830}

864FlagStatus flag_parse_and_invoke(FlagParser* fp, int argc, char** argv, void* user_data) {
865 if (!fp || argc < 1) {
866 return FLAG_ERROR_INVALID_ARGUMENT;
867 }
868
869 // Parse all arguments
870 FlagStatus status = flag_parse(fp, argc, argv);
871 if (status != FLAG_OK) {
872 return status;
873 }
874
875 // Run pre-invocation callback (Global setup)
876 if (fp->pre_invoke) {
877 fp->pre_invoke(user_data);
878 }
879
880 // Find the deepest active subcommand (The "Leaf" command)
881 FlagParser* target = fp;
882 while (target->active_subcommand) {
883 target = target->active_subcommand;
884 }
885
886 // Run the handler for the target command
887 if (target->handler) {
888 target->handler(user_data);
889 }
890
891 return FLAG_OK;
892}

941static void print_usage_rec(FlagParser* fp, bool is_sub) {
942 if (!fp) return;
943
944 printf("Usage: %s%s [flags] %s\n", fp->name, is_sub ? " <subcmd>" : "", fp->cmd_count > 0 ? "<command>" : "[args]");
945
946 if (fp->description) printf("\n%s\n", fp->description);
947
948 // Calculate alignment
949 size_t max_width = 0;
950 for (size_t i = 0; i < fp->flag_count; i++) {
951 size_t w = 6; // " -x, "
952 w += strlen(fp->flags[i].name) + 2; // --name
953 if (fp->flags[i].type != TYPE_BOOL) {
954 w += strlen(type_to_str(fp->flags[i].type)) + 1; // =TYPE
955 }
956 if (w > max_width) max_width = w;
957 }
958
959 if (fp->flag_count > 0) printf("\nFlags:\n");
960 for (size_t i = 0; i < fp->flag_count; i++) {
961 Flag* f = &fp->flags[i];
962
963 // Build the left column: " -s, --long=TYPE"
964 char left[128];
965 int pos = 0;
966 pos += snprintf(left + pos, (size_t)(128 - pos), " ");
967 if (f->short_name)
968 pos += snprintf(left + pos, (size_t)(128 - pos), "-%c, ", f->short_name);
969 else
970 pos += snprintf(left + pos, (size_t)(128 - pos), " ");
971
972 pos += snprintf(left + pos, (size_t)(128 - pos), "--%s", f->name);
973 if (f->type != TYPE_BOOL) {
974 pos += snprintf(left + pos, (size_t)(128 - pos), "=%s", type_to_str(f->type));
975 }
976
977 // Print row
978 printf("%-*s %s", (int)max_width, left, f->description ? f->description : "");
979 if (f->required) {
980 printf(" (Required)");
