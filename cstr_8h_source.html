630static inline char* trim_string(char* str) {
631 if (!str) return NULL;
632
633 // 1. Trim leading whitespace
634 while (isspace((unsigned char)*str)) {
635 str++;
636 }
637
638 // Optimization: If string is empty or all spaces, return early
639 if (*str == '\0') {
640 return str;
641 }
642
643 // 2. Single-pass forward scan for trailing trim
644 // 'cursor' finds the null terminator.
645 // 'end' tracks the last seen non-space character.
646 char* end = str;
647 char* cursor = str;
648
649 while (*cursor) {
650 if (!isspace((unsigned char)*cursor)) {
651 end = cursor;
652 }
653 cursor++;
654 }
655
656 // 3. Terminate after the last non-space character
657 *(end + 1) = '\0';
658
659 return str;
660}

674static inline char* strcasestr(const char* haystack, const char* needle) {
675 if (needle == NULL || *needle == '\0') {
676 return (char*)haystack;
677 }
678
679 const size_t needle_len = strlen(needle);
680 const size_t haystack_len = strlen(haystack);
681
682 if (needle_len > haystack_len) {
683 return NULL;
684 }
685
686 const size_t search_len = haystack_len - needle_len + 1;
687
688 for (size_t i = 0; i < search_len; i++) {
689 if (tolower((unsigned char)haystack[i]) == tolower((unsigned char)needle[0])) {
690 bool match = true;
691 for (size_t j = 1; j < needle_len; j++) {
692 if (tolower((unsigned char)haystack[i + j]) != tolower((unsigned char)needle[j])) {
693 match = false;
694 break;
695 }
696 }
697 if (match) {
698 return (char*)(haystack + i);
699 }
700 }
701 }
702
703 return NULL;
704}

713static inline int strcasecmp(const char* s1, const char* s2) {
714 if (s1 == NULL || s2 == NULL) {
715 // Handle NULL: NULL == NULL returns 0, otherwise compare addresses
716 if (s1 == s2) return 0;
717 return (s1 == NULL) ? -1 : 1;
718 }
719
720 while (*s1 != '\0' && *s2 != '\0') {
721 const int c1 = tolower((unsigned char)*s1);
722 const int c2 = tolower((unsigned char)*s2);
723
724 if (c1 != c2) {
725 return c1 - c2;
726 }
727
728 s1++;
729 s2++;
730 }
731
732 // If we've reached here, one or both strings ended
733 return tolower((unsigned char)*s1) - tolower((unsigned char)*s2);
734}

744static inline int strncasecmp(const char* s1, const char* s2, size_t n) {
745 if (n == 0) {
746 return 0;
747 }
748
749 if (s1 == NULL || s2 == NULL) {
750 if (s1 == s2) return 0;
751 return (s1 == NULL) ? -1 : 1;
752 }
753
754 for (size_t i = 0; i < n; i++) {
755 const int c1 = tolower((unsigned char)s1[i]);
756 const int c2 = tolower((unsigned char)s2[i]);
757
758 if (c1 != c2) {
759 return c1 - c2;
760 }
761
762 // If we hit null terminator in either string, we're done
763 if (s1[i] == '\0') {
764 return 0;
765 }
766 }
767
768 return 0; // First n characters match
769}
