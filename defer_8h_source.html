87// 1. The RAII Holder that runs the lambda on destruction
88template <typename F>
89struct Deferrer {
90 F f;
91 Deferrer(F&& func) : f(std::forward<F>(func)) {}
92 ~Deferrer() { f(); }
93 Deferrer(const Deferrer&) = delete;
