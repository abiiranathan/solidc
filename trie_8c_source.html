104bool trie_insert(trie_t* trie, const char* word) {
105 if (trie == NULL || word == NULL || *word == '\0') {

107 }
108
109 trie_node_t* current = trie->root;
110 const unsigned char* uword = (const unsigned char*)word;
111
112 // Traverse/create path for each character
113 while (*uword != '\0') {
114 unsigned char index = *uword;
115
116 if (current->children[index] == NULL) {
117 current->children[index] = trie_node_create();
118 if (current->children[index] == NULL) {
119 return false; // Allocation failure
120 }
121 }
122
123 current = current->children[index];
124 uword++;
125 }
126
127 // Mark end of word and increment frequency

129 current->is_end_of_word = true;
130 trie->word_count++;
131 }
132 current->frequency++;
133
134 return true;

276static void collect_words_recursive(const trie_node_t* node, char* prefix, size_t max_len, size_t depth,
277 suggestions_collector_t* collector, Arena* arena) {
278 if (node == NULL || collector->count >= collector->limit) {
279 return;
280 }
281
282 // If this node marks end of word, add it to suggestions
283 if (node->is_end_of_word) {
284 prefix[depth] = '\0';
285
286 // Duplicate string into the Arena
287 char* dup = arena_strdup(arena, prefix);
288 if (dup != NULL) {
289 collector->suggestions[collector->count++] = dup;
290 }
291 }
292
293 // Recursively traverse all children
294 for (size_t i = 0; i < TRIE_CHARSET_SIZE; i++) {
295 if (node->children[i] != NULL) {
296 // Check buffer bounds
297 if (depth + 1 >= max_len) {
298 return; // Prevent buffer overflow
299 }
300
301 prefix[depth] = (char)i;
302 collect_words_recursive(node->children[i], prefix, max_len, depth + 1, collector, arena);
303
304 if (collector->count >= collector->limit) {
305 return; // Reached limit, stop collecting
306 }
307 }
308 }
309}

323const char** trie_autocomplete(const trie_t* trie, const char* prefix, size_t max_suggestions, size_t* out_count,
324 Arena* arena) {
325 if (out_count != NULL) {
326 *out_count = 0;
327 }
328
329 if (trie == NULL || prefix == NULL || out_count == NULL || max_suggestions == 0 || arena == NULL) {
330 return NULL;
331 }
332
333 // Navigate to the prefix node
334 const trie_node_t* current = trie->root;
335 const unsigned char* uprefix = (const unsigned char*)prefix;
336 size_t prefix_len = strlen(prefix);
337
338 while (*uprefix != '\0') {
339 unsigned char index = *uprefix;
340 if (current->children[index] == NULL) {
341 return NULL; // Prefix doesn't exist
342 }
343 current = current->children[index];
344 uprefix++;
345 }
346
347 // Initialize collector
348 // We allocate the full limit upfront in the Arena to avoid resizing logic
349 suggestions_collector_t collector;
350 collector.limit = max_suggestions;
351 collector.count = 0;
352 collector.suggestions = (char**)ARENA_ALLOC_ARRAY(arena, char*, max_suggestions);
353
354 if (collector.suggestions == NULL) {
355 return NULL;
356 }
357
358 // Allocate buffer for building words in the Arena
359 const size_t max_word_len = 1024;
360 char* word_buffer = (char*)arena_alloc(arena, max_word_len);
361
362 if (word_buffer == NULL) {
363 return NULL;
364 }
365
366 // Copy prefix into buffer
367 if (prefix_len >= max_word_len) {
368 return NULL;
369 }
370 strncpy(word_buffer, prefix,
371 prefix_len + 1); // +1 to copy null terminator if space allows, mostly to silence warnings
372
373 // Collect all words with this prefix
374 collect_words_recursive(current, word_buffer, max_word_len, prefix_len, &collector, arena);
375
376 // Return results
377 if (collector.count == 0) {
378 return NULL;
379 }
