64void ucp_to_utf8(uint32_t codepoint, char utf8[UTF8_MAX_LEN]) {
65 if (!utf8) { return; }
66
67 if (codepoint <= 0x7F) {
68 utf8[0] = (char)codepoint;
69 utf8[1] = '\0';
70 } else if (codepoint <= 0x7FF) {
71 utf8[0] = (char)(0xC0 | (codepoint >> 6));
72 utf8[1] = (char)(0x80 | (codepoint & 0x3F));
73 utf8[2] = '\0';
74 } else if (codepoint <= 0xFFFF) {
75 utf8[0] = (char)(0xE0 | (codepoint >> 12));
76 utf8[1] = (char)(0x80 | ((codepoint >> 6) & 0x3F));
77 utf8[2] = (char)(0x80 | (codepoint & 0x3F));
78 utf8[3] = '\0';
79 } else if (codepoint <= UNICODE_MAX_CODEPOINT) {
80 utf8[0] = (char)(0xF0 | (codepoint >> 18));
81 utf8[1] = (char)(0x80 | ((codepoint >> 12) & 0x3F));
82 utf8[2] = (char)(0x80 | ((codepoint >> 6) & 0x3F));
83 utf8[3] = (char)(0x80 | (codepoint & 0x3F));
84 utf8[4] = '\0';
85 } else {
86 /* Codepoint out of range: produce an empty string rather than garbage. */
87 utf8[0] = '\0';

111uint32_t utf8_to_codepoint(const char* utf8) {
112 if (!utf8) {
113 return 0xFFFD; // replacement character
114 }
115
116 uint32_t codepoint = 0;
117 const uint8_t* u = (const uint8_t*)utf8;
118
119 if ((u[0] & 0x80) == 0) {
120 /* 1-byte ASCII sequence. */
121 codepoint = u[0];
122 } else if ((u[0] & 0xE0) == 0xC0) {
123 /* 2-byte sequence. */
124 if ((u[1] & 0xC0) == 0x80) {
125 codepoint = ((u[0] & 0x1FU) << 6) | (u[1] & 0x3F);
126 if (codepoint < 0x80) {
127 return 0xFFFD; // overlong encoding
128 }
129 } else {
130 return 0xFFFD;
131 }
132 } else if ((u[0] & 0xF0) == 0xE0) {
133 /* 3-byte sequence. */
134 if ((u[1] & 0xC0) == 0x80 && (u[2] & 0xC0) == 0x80) {
135 codepoint = ((u[0] & 0x0FU) << 12) | ((u[1] & 0x3FU) << 6) | (u[2] & 0x3F);
136 if (codepoint < 0x800) {
137 return 0xFFFD; // overlong encoding
138 }
139 if (codepoint >= 0xD800 && codepoint <= 0xDFFF) {
140 return 0xFFFD; // surrogate halves are invalid in UTF-8
141 }
142 } else {
143 return 0xFFFD;
144 }
145 } else if ((u[0] & 0xF8) == 0xF0) {
146 /* 4-byte sequence. */
147 if ((u[1] & 0xC0) == 0x80 && (u[2] & 0xC0) == 0x80 && (u[3] & 0xC0) == 0x80) {
148 codepoint = ((u[0] & 0x07U) << 18) | ((u[1] & 0x3FU) << 12) | ((u[2] & 0x3FU) << 6) | (u[3] & 0x3F);
149 if (codepoint < 0x10000) {
150 return 0xFFFD; // overlong encoding
151 }
152 if (codepoint > UNICODE_MAX_CODEPOINT) { return 0xFFFD; }
153 } else {
154 return 0xFFFD;
155 }
156 } else {
157 /* Lone continuation byte or invalid leading byte (0xF8-0xFF). */
158 return 0xFFFD;
159 }
160
161 return codepoint;
162}

304bool is_valid_utf8(const char* utf8) {
305 if (!utf8) { return false; }
306
307 for (size_t i = 0; utf8[i] != '\0';) {
308 unsigned char byte = (unsigned char)utf8[i];
309 if ((byte & 0x80) == 0) {
310 i++;
311 } else if ((byte & 0xE0) == 0xC0) {
312 if (utf8[i + 1] == '\0' || ((unsigned char)utf8[i + 1] & 0xC0) != 0x80) { return false; }
313 uint32_t codepoint = ((uint32_t)(byte & 0x1F) << 6) | ((unsigned char)utf8[i + 1] & 0x3F);
314 if (codepoint < 0x80) {
315 return false; // overlong encoding
316 }
317 i += 2;
318 } else if ((byte & 0xF0) == 0xE0) {
319 if (utf8[i + 1] == '\0' || utf8[i + 2] == '\0' || ((unsigned char)utf8[i + 1] & 0xC0) != 0x80 ||
320 ((unsigned char)utf8[i + 2] & 0xC0) != 0x80) {
321 return false;
322 }
323 uint32_t codepoint = ((uint32_t)(byte & 0x0F) << 12) |
324 ((uint32_t)((unsigned char)utf8[i + 1] & 0x3F) << 6) |
325 ((unsigned char)utf8[i + 2] & 0x3F);
326 if (codepoint < 0x800 || (codepoint >= 0xD800 && codepoint <= 0xDFFF)) {
327 return false; // overlong encoding or surrogate half
328 }
329 i += 3;
330 } else if ((byte & 0xF8) == 0xF0) {
331 if (utf8[i + 1] == '\0' || utf8[i + 2] == '\0' || utf8[i + 3] == '\0' ||
332 ((unsigned char)utf8[i + 1] & 0xC0) != 0x80 || ((unsigned char)utf8[i + 2] & 0xC0) != 0x80 ||
333 ((unsigned char)utf8[i + 3] & 0xC0) != 0x80) {
334 return false;
335 }
336 uint32_t codepoint = ((uint32_t)(byte & 0x07) << 18) |
337 ((uint32_t)((unsigned char)utf8[i + 1] & 0x3F) << 12) |
338 ((uint32_t)((unsigned char)utf8[i + 2] & 0x3F) << 6) |
339 ((unsigned char)utf8[i + 3] & 0x3F);
340 if (codepoint < 0x10000 || codepoint > UNICODE_MAX_CODEPOINT) {
341 return false; // overlong encoding or out of range
342 }
343 i += 4;
344 } else {
345 return false; // lone continuation byte or invalid leading byte
346 }
347 }
348 return true;
349}

499static utf8_string* utf8_string_alloc(size_t data_capacity) {
500 /* Guard against overflow: sizeof(utf8_string) + data_capacity + 1 must
501 * not wrap around on platforms with a small size_t. */
502 if (data_capacity > SIZE_MAX - sizeof(utf8_string) - 1) { return NULL; }
503
504 utf8_string* s = (utf8_string*)malloc(sizeof(*s) + data_capacity + 1);
505 if (!s) { return NULL; }
506
507 s->data = (char*)s + sizeof(*s);
508 s->data[0] = '\0';
509 s->length = 0;
510 s->count = 0;
511 s->capacity = data_capacity;
512 return s;
513}

537static bool utf8_string_grow(utf8_string** s_ptr, size_t min_capacity) {
538 utf8_string* s = *s_ptr;
539 if (s->capacity >= min_capacity) {
540 return true; // already large enough
541 }
542
543 size_t new_capacity = s->capacity * 2;
544 if (new_capacity < min_capacity) { new_capacity = min_capacity; }
545
546 if (new_capacity > SIZE_MAX - sizeof(*s) - 1) {
547 return false; // would overflow the allocation size computation
548 }
549
550 utf8_string* new_s = (utf8_string*)realloc(s, sizeof(*s) + new_capacity + 1);
551 if (!new_s) {
552 return false; // *s_ptr unchanged; original block still valid
553 }
554
555 new_s->data = (char*)new_s + sizeof(*new_s);
556 new_s->capacity = new_capacity;
557 *s_ptr = new_s;
558 return true;
559}

748int utf8_last_index_of(const utf8_string* s, const char* utf8) {
749 if (!s || !s->data || !utf8) { return -1; }
750
751 size_t needle_len = utf8_valid_byte_count(utf8);
752 if (needle_len == 0 || needle_len > s->length) { return -1; }
753
754 /* i is the highest byte offset at which the needle could still fit;
755 * scan downward so the first match found is the last occurrence. */
756 size_t i = s->length - needle_len;
757 for (;;) {
758 if (memcmp(&s->data[i], utf8, needle_len) == 0) { return (int)i; }
759 if (i == 0) { break; }
760 i--;
761 }
762 return -1;
763}

961bool utf8_remove(utf8_string* s, size_t index, size_t count) {
962 if (!s || !s->data || index >= s->length || count == 0) { return false; }
963
964 size_t i = index;
965 size_t removed = 0; // actual codepoints removed, may be less than `count`
966 for (; removed < count && i < s->length; removed++) {
967 size_t len = utf8_char_length(&s->data[i]);
968 if (len == 0) {
969 break; // malformed byte; stop removing here
970 }
971 i += len;
972 }
973
974 if (i > s->length) { i = s->length; }
975
976 memmove(&s->data[index], &s->data[i], s->length - i + 1);
977 s->length -= (i - index);
978 s->count -= removed;
979 return true;
980}

1016 memcpy(&s->data[offset], new_str, new_byte_len);
1017 s->length = s->length - old_byte_len + new_byte_len;
1018 s->count = s->count - old_count + new_count;

1052size_t utf8_replace_all(utf8_string** s_ptr, const char* old_str, const char* new_str) {

1058 }
1059

1061 size_t old_count = utf8_count_codepoints(old_str);

1070 if (new_byte_len > old_byte_len) {
1071 /* Growing: may need more capacity, which may relocate the block. */
1072 if (!utf8_string_grow(s_ptr, s->length - old_byte_len + new_byte_len)) {
1073 return replacements; // stop here; *s_ptr still valid for prior replacements
1074 }
1075 s = *s_ptr; // utf8_string_grow() may have relocated the block
1076 }
1077 cursor = s->data + offset; // re-derive: s->data may differ even without growth
1078
1079 memmove(&s->data[offset + new_byte_len], &s->data[offset + old_byte_len],
1080 s->length - offset - old_byte_len + 1);
1081 memcpy(&s->data[offset], new_str, new_byte_len);
1082 s->length = s->length - old_byte_len + new_byte_len;
1083 s->count = s->count - old_count + new_count;
1084
1085 cursor += new_byte_len;
1086 replacements++;
1087 }
1088
1089 return replacements;
1090}

1266void utf8_trim_chars(char* str, const char* chars) {
1267 if (!str || !chars) { return; }
1268
1269 enum { MAX_TRIM_CODEPOINTS = 256 };
1270 uint32_t trim_codepoints[MAX_TRIM_CODEPOINTS];
1271 size_t num_trim_chars = 0;
1272 size_t chars_len = strlen(chars);
1273
1274 /* Build the de-duplicated set of codepoints to trim. */
1275 for (size_t k = 0; k < chars_len && num_trim_chars < MAX_TRIM_CODEPOINTS;) {
1276 size_t current_char_len = utf8_char_length(&chars[k]);
1277 if (current_char_len == 0 || k + current_char_len > chars_len) {
1278 k++;
1279 continue;
1280 }
1281
1282 uint32_t codepoint = utf8_to_codepoint(&chars[k]);
1283 if (codepoint != 0xFFFD) {
1284 bool found = false;
1285 for (size_t idx = 0; idx < num_trim_chars; idx++) {
1286 if (trim_codepoints[idx] == codepoint) {
1287 found = true;
1288 break;
1289 }
1290 }
1291 if (!found && num_trim_chars < MAX_TRIM_CODEPOINTS) { trim_codepoints[num_trim_chars++] = codepoint; }
1292 }
1293 k += current_char_len;
1294 }
1295
1296 size_t len = strlen(str);
1297
1298 /* Trim from the front. */
1299 size_t i = 0;
1300 while (i < len) {
1301 size_t current_char_len = utf8_char_length(&str[i]);
1302 if (current_char_len == 0 || i + current_char_len > len) { break; }
1303
1304 uint32_t codepoint = utf8_to_codepoint(&str[i]);
1305 bool should_trim = false;
1306 if (codepoint != 0xFFFD) {
1307 for (size_t j = 0; j < num_trim_chars; j++) {
1308 if (codepoint == trim_codepoints[j]) {
1309 should_trim = true;
1310 break;
1311 }
1312 }
1313 }
1314
1315 if (!should_trim) { break; }
1316 i += current_char_len;
1317 }
1318
1319 if (i > 0) {
1320 memmove(str, &str[i], len - i + 1);
1321 len -= i;
1322 }
1323
1324 /* Trim from the back. */
1325 i = len;
1326 while (i > 0) {
1327 size_t char_start = i - 1;
1328 while (char_start > 0 && ((unsigned char)str[char_start] & 0xC0) == 0x80) {
1329 char_start--;
1330 }
1331
1332 if (((unsigned char)str[char_start] & 0xC0) == 0x80) {
1333 /* Could not find a leading byte; the buffer is malformed near
1334 * its end. Stop rather than misinterpret a continuation byte
1335 * as a character start. */
1336 break;
1337 }
1338
1339 size_t char_len = utf8_char_length(&str[char_start]);
1340 if (char_len == 0 || char_start + char_len != i) { break; }
1341
1342 uint32_t codepoint = utf8_to_codepoint(&str[char_start]);
1343 bool should_trim = false;
1344 if (codepoint != 0xFFFD) {
1345 for (size_t j = 0; j < num_trim_chars; j++) {
1346 if (codepoint == trim_codepoints[j]) {
1347 should_trim = true;
1348 break;
1349 }
1350 }
1351 }
1352
1353 if (!should_trim) { break; }
1354 i = char_start;
1355 }
1356
1357 str[i] = '\0';
1358}

1403void utf8_tolower(char* str) {
1404 if (!str) { return; }
1405
1406 for (size_t i = 0; str[i] != '\0';) {
1407 unsigned char byte = (unsigned char)str[i];
1408
1409 /* ---- ASCII fast path (covers the vast majority of real text) ---- */
1410 if (byte < 0x80u) {
1411 /* Branchless lowercase: set bit 5 only when byte is 'A'..'Z'. */
1412 unsigned int is_upper = (unsigned int)(byte - 'A' + 1u <= 26u);
1413 str[i] = (char)(byte | (is_upper << 5));
1414 i++;
1415 continue;
1416 }
1417
1418 /* ---- Multibyte path ---- */
1419 uint32_t codepoint = utf8_to_codepoint(&str[i]);
1420 size_t old_len = utf8_char_length(&str[i]);
1421 if (old_len == 0) {
1422 break; // malformed byte; stop rather than loop on it
1423 }
1424
1425#if WCHAR_MAX < UNICODE_MAX_CODEPOINT
1426 if (codepoint > (uint32_t)WCHAR_MAX) {
1427 i += old_len; // outside what towlower() can safely accept here
1428 continue;
1429 }
1430#endif
1431
1432 if (iswupper((wint_t)codepoint)) {
1433 char utf8[UTF8_MAX_LEN] = {0};
1434 ucp_to_utf8((uint32_t)towlower((wint_t)codepoint), utf8);
1435 size_t new_len = utf8_valid_byte_count(utf8);
1436
1437 if (new_len != old_len) {
1438 size_t remaining = strlen(&str[i + old_len]);
1439 memmove(&str[i + new_len], &str[i + old_len], remaining + 1);
1440 }
1441 memcpy(&str[i], utf8, new_len);
1442 i += new_len;
1443 } else {
1444 i += old_len;
1445 }
1446 }
1447}

1527utf8_string** utf8_split(const utf8_string* str, const char* delim, size_t* num_parts) {
1528 if (!num_parts) { return NULL; }
1529 *num_parts = 0;
1530
1531 if (!str || !str->data || !delim) { return NULL; }
1532
1533 size_t delim_len = utf8_valid_byte_count(delim);
1534 if (delim_len == 0) { return NULL; }
1535
1536 /* First pass: count how many parts the split will produce. */
1537 size_t count = 1;
1538 size_t len = str->length;
1539 for (size_t i = 0; i < len;) {
1540 if (i + delim_len <= len && memcmp(&str->data[i], delim, delim_len) == 0) {
1541 count++;
1542 i += delim_len;
1543 } else {
1544 size_t char_len = utf8_char_length(&str->data[i]);
1545 if (char_len == 0) {
1546 break; // malformed byte; stop scanning, finish with what we have
1547 }
1548 i += char_len;
1549 }
1550 }
1551
1552 utf8_string** parts = (utf8_string**)malloc(count * sizeof(*parts));
1553 if (!parts) { return NULL; }
1554
1555 /* Second pass: actually carve out each part. utf8_new() always copies a
1556 * NUL-terminated string, so we build each part from a temporary
1557 * substring rather than mutating str->data in place. */
1558 size_t index = 0;
1559 size_t start = 0;
1560 for (size_t i = 0; i < len;) {
1561 if (i + delim_len <= len && memcmp(&str->data[i], delim, delim_len) == 0) {
1562 char* piece = (char*)malloc(i - start + 1);
1563 if (!piece) { goto split_alloc_failed; }
1564 memcpy(piece, &str->data[start], i - start);
1565 piece[i - start] = '\0';
1566
1567 parts[index] = utf8_new(piece);
1568 free(piece);
1569 if (!parts[index]) { goto split_alloc_failed; }
1570 index++;
1571
1572 i += delim_len;
1573 start = i;
1574 } else {
1575 size_t char_len = utf8_char_length(&str->data[i]);
1576 if (char_len == 0) { break; }
1577 i += char_len;
1578 }
1579 }
1580
1581 /* Final trailing part, from `start` to the end of the string. */
1582 {
1583 char* piece = (char*)malloc(len - start + 1);
1584 if (!piece) { goto split_alloc_failed; }
1585 memcpy(piece, &str->data[start], len - start);
1586 piece[len - start] = '\0';
1587
1588 parts[index] = utf8_new(piece);
1589 free(piece);
1590 if (!parts[index]) { goto split_alloc_failed; }
1591 index++;
1592 }
1593
1594 *num_parts = index;
1595 return parts;
1596
1597split_alloc_failed:
1598 /* Clean up every part successfully built so far, then the array itself,
1599 * so the caller never has to special-case a partially-filled result. */
1600 for (size_t k = 0; k < index; k++) {
1601 utf8_free(parts[k]);
1602 }
1603 free(parts);
1604 *num_parts = 0;
1605 return NULL;
1606}
