58/*
59 * Tag byte encoding (8 bits):
60 *
61 * bit 7 (0x80) : TAG_EMPTY — slot never written
62 * bit 6 (0x40) : TAG_DELETED — tombstone (slot was occupied, now free)
63 * bits 5-0 : 6-bit hash fragment for fast reject
64 *
65 * Invariant: a live slot has both control bits clear (tag & 0xC0 == 0).
66 * Using bit 7 for EMPTY and bit 6 for DELETED means a single AND with 0xC0
67 * distinguishes all three states with no branch.
68 *
69 * The 6-bit fragment is derived from the full 32-bit hash (bits 8-13, chosen
70 * to avoid overlap with the bits used for bucket index selection). Because
71 * bucket_count is always a power of 2 and INITIAL_BUCKET_MULTIPLIER=2, the
72 * bucket index uses at most log2(131072)=17 low bits. Taking fragment bits
73 * from bits 18-23 gives maximum independence from the index selection.
74 */
75#define TAG_EMPTY 0x80u /* bit 7: slot never used */
76#define TAG_DELETED 0x40u /* bit 6: tombstone */
77#define TAG_CONTROL_MASK 0xC0u /* bits 7:6 — any set means not live */
78#define TAG_FRAGMENT_SHIFT 18u /* start bit for the 6-bit fragment */
79#define TAG_FRAGMENT_MASK 0x3Fu /* 6-bit fragment mask */

108static inline void probe_stat_record(size_t probes) {
109 if (unlikely(probes == 0)) return;
110 size_t bucket = (probes - 1) < PROBE_STAT_BUCKETS ? (probes - 1) : PROBE_STAT_BUCKETS - 1;
111 tl_probe_hist[bucket]++;
112 tl_probe_total_ops++;

119void cache_probe_stats_flush(void) {
120 for (size_t i = 0; i < PROBE_STAT_BUCKETS; i++) {
121 atomic_fetch_add_explicit(&g_probe_hist[i], tl_probe_hist[i], memory_order_relaxed);
122 tl_probe_hist[i] = 0;
123 }
124 atomic_fetch_add_explicit(&g_probe_total_ops, tl_probe_total_ops, memory_order_relaxed);
125 tl_probe_total_ops = 0;
126}

210static inline size_t next_power_of_2(size_t n) {
211 if (n && !(n & (n - 1))) return n;
212 n--;
213 n |= n >> 1;
214 n |= n >> 2;
215 n |= n >> 4;
216 n |= n >> 8;
217 n |= n >> 16;
218 n |= n >> 32; /* Handle 64-bit size constraints */
219 return n + 1;
220}

225static inline uint32_t hash_key(const char* key, size_t len) {
226 uint32_t hash = 2166136261u; /* FNV offset basis */
227 const unsigned char* p = (const unsigned char*)key;
228 const unsigned char* end = p + len;
229 while (p < end) {
230 hash ^= *p++;
231 hash *= 16777619u; /* FNV prime */
232 }
233 /* Reserve 0 and 1 as historical control values */
234 if (unlikely(hash < 2u)) return hash + 2u;
235 return hash;
236}

289static inline bool keys_equal(const void* k1, const void* k2, size_t len) {
290 if (len == 8) {
291 uint64_t u1, u2;
292 memcpy(&u1, k1, 8);
293 memcpy(&u2, k2, 8);
294 return u1 == u2;
295 }
296 if (len == 4) {
297 uint32_t u1, u2;
298 memcpy(&u1, k1, 4);
299 memcpy(&u2, k2, 4);
300 return u1 == u2;
301 }
302 if (len == 16) {
303 uint64_t u1[2], u2[2];
304 memcpy(u1, k1, 16);
305 memcpy(u2, k2, 16);
306 return u1[0] == u2[0] && u1[1] == u2[1];
307 }
308 if (len == 2) {
309 uint16_t u1, u2;
310 memcpy(&u1, k1, 2);
311 memcpy(&u2, k2, 2);
312 return u1 == u2;
313 }
314 if (len == 1) { return *(const uint8_t*)k1 == *(const uint8_t*)k2; }
315 return memcmp(k1, k2, len) == 0;
316}

323static size_t find_slot(aligned_cache_shard_t* shard, uint32_t hash, const char* key, size_t klen, uint8_t target_tag,
324 bool* found) {
325 size_t mask = shard->bucket_count - 1;
326 size_t idx = hash & mask; /* initial probe position */
327 size_t first_tombstone = SIZE_MAX;
328 uint8_t* tags = shard->tags;
329 cache_entry_t** entries = shard->entries;
330
331 *found = false;
332
333 PROBE_INIT(probes);
334
335 for (size_t probe_count = 0; probe_count < shard->bucket_count; probe_count++) {
336 uint8_t tag = tags[idx];
337 PROBE_INC(probes);
338
339 /*
340 * Since target_tag is built by masking with 0x3F, it can never have bits 6 or 7 set.
341 * Therefore, any tag exactly matching target_tag is guaranteed to be a live entry.
342 */
343 if (likely(tag == target_tag)) {
344 cache_entry_t* entry = entries[idx];
345
346 /* Pack hash:32 and key_len:32 into a single 64-bit load/comparison */
347 uint64_t entry_meta;
348 memcpy(&entry_meta, &entry->hash, 8);
349 uint64_t target_meta = ((uint64_t)klen << 32) | hash;
350
351 if (likely(entry_meta == target_meta) && keys_equal(entry->data, key, klen)) {
352 *found = true;
353 PROBE_RECORD(probes);
354 return idx;
355 }
356 } else if (tag & TAG_EMPTY) {
357 /* Empty slot terminates the probe chain */
358 PROBE_RECORD(probes);
359 return (first_tombstone != SIZE_MAX) ? first_tombstone : idx;
360 } else if (tag & TAG_DELETED) {
361 /* Tombstone: keep probing, record for potential reuse. */
362 if (first_tombstone == SIZE_MAX) first_tombstone = idx;
363 }
364
365 idx = (idx + 1) & mask; /* advance: linear probing with power-of-2 wrap */
366 }
367
368 /* Table fully probed (should not happen under normal operation). */
369 PROBE_RECORD(probes);
370 return (first_tombstone != SIZE_MAX) ? first_tombstone : idx;
371}

379static void compact_shard(aligned_cache_shard_t* shard) {
380 size_t n = shard->bucket_count;
381
382 uint8_t* new_tags = ALIGNED_ALLOC(CACHE_LINE_SIZE, n * sizeof(uint8_t));
383 if (!new_tags) return;
384
385 cache_entry_t** new_entries = ALIGNED_ALLOC(CACHE_LINE_SIZE, n * sizeof(cache_entry_t*));
386 if (!new_entries) {
387 free(new_tags);
388 return;
389 }
390
391#if defined(__linux__) && defined(MADV_HUGEPAGE)
392 madvise(new_tags, n * sizeof(uint8_t), MADV_HUGEPAGE);
393 madvise(new_entries, n * sizeof(cache_entry_t*), MADV_HUGEPAGE);
394#endif
395
396 /* Initialize all slots to EMPTY. */
397 memset(new_tags, TAG_EMPTY, n * sizeof(uint8_t));
398 memset(new_entries, 0, n * sizeof(cache_entry_t*));
399
400 size_t mask = n - 1;
401 uint8_t* old_tags = shard->tags;
402 cache_entry_t** old_entries = shard->entries;
403
404 for (size_t i = 0; i < n; i++) {
405 uint8_t tag = old_tags[i];
406 if (tag & TAG_CONTROL_MASK) continue;
407
408 cache_entry_t* entry = old_entries[i];
409 uint32_t h = entry->hash;
410 uint8_t ntag = make_tag(h);
411 size_t idx = h & mask;
412
413 while (!(new_tags[idx] & TAG_EMPTY)) {
414 idx = (idx + 1) & mask;
415 }
416 new_tags[idx] = ntag;
417 new_entries[idx] = entry;
418 }
419
420 shard->tags = new_tags;
421 shard->entries = new_entries;
422 shard->tombstone_count = 0;
423 shard->clock_hand = 0;
424
425 free(old_tags);
426 free(old_entries);
427}

434static bool clock_evict(aligned_cache_shard_t* shard) {
435 size_t scanned = 0;
436 size_t mask = shard->bucket_count - 1;
437 uint8_t* tags = shard->tags;
438 cache_entry_t** entries = shard->entries;
439
440 while (scanned < shard->bucket_count) {
441 size_t idx = shard->clock_hand;
442 shard->clock_hand = (shard->clock_hand + 1) & mask;
443
444 uint8_t tag = tags[idx];
445 if (tag & TAG_CONTROL_MASK) {
446 scanned++;
447 continue;
448 }
449
450 cache_entry_t* entry = entries[idx];
451 uint8_t bit = atomic_load_explicit(&entry->clock_bit, memory_order_relaxed);
452
453 if (bit == 0) {
454 tags[idx] = TAG_DELETED;
455 entries[idx] = NULL;
456 shard->size--;
457 shard->tombstone_count++;
458 entry_ref_dec(entry);
459 return true;
460 }
461
462 atomic_store_explicit(&entry->clock_bit, 0, memory_order_relaxed);
463 scanned++;
464 }
465
466 for (size_t i = 0; i < shard->bucket_count; i++) {
467 size_t idx = shard->clock_hand;
468 shard->clock_hand = (shard->clock_hand + 1) & mask;
469 if (!(tags[idx] & TAG_CONTROL_MASK)) {
470 cache_entry_t* entry = entries[idx];
471 tags[idx] = TAG_DELETED;
472 entries[idx] = NULL;
473 shard->size--;
474 shard->tombstone_count++;
475 entry_ref_dec(entry);
476 return true;
477 }
478 }
479 return false;
480}

487cache_t* cache_create(size_t capacity, uint32_t default_ttl) {
488 struct cache_s* c = calloc(1, sizeof(struct cache_s));
489 if (!c) return NULL;
490
491 if (capacity == 0) capacity = 1000;
492 size_t shard_cap = capacity / CACHE_SHARD_COUNT;
493 if (shard_cap < 1) shard_cap = 1;
494
495 c->default_ttl = default_ttl ? default_ttl : CACHE_DEFAULT_TTL;
496
497 for (size_t i = 0; i < CACHE_SHARD_COUNT; i++) {
498 aligned_cache_shard_t* s = &c->shards[i];
499 s->capacity = shard_cap;
500
501 size_t desired = (size_t)(shard_cap * INITIAL_BUCKET_MULTIPLIER);
502 s->bucket_count = next_power_of_2(desired);
503
504 s->tags = ALIGNED_ALLOC(CACHE_LINE_SIZE, s->bucket_count * sizeof(uint8_t));
505 if (!s->tags) goto cleanup_error;
506
507 s->entries = ALIGNED_ALLOC(CACHE_LINE_SIZE, s->bucket_count * sizeof(cache_entry_t*));
508 if (!s->entries) {
509 free(s->tags);
510 s->tags = NULL;
511 goto cleanup_error;
512 }
513
514#if defined(__linux__) && defined(MADV_HUGEPAGE)
515 madvise(s->tags, s->bucket_count * sizeof(uint8_t), MADV_HUGEPAGE);
516 madvise(s->entries, s->bucket_count * sizeof(cache_entry_t*), MADV_HUGEPAGE);
517#endif
518 memset(s->tags, TAG_EMPTY, s->bucket_count * sizeof(uint8_t));
519 memset(s->entries, 0, s->bucket_count * sizeof(cache_entry_t*));
520
521 fast_rwlock_init(&s->lock);
522 }
523
524 return (cache_t*)c;
525
526cleanup_error:
527 for (size_t j = 0; j < CACHE_SHARD_COUNT; j++) {
528 free(c->shards[j].tags);
529 free(c->shards[j].entries);
530 }
531 free(c);
532 return NULL;
533}

538void cache_destroy(cache_t* cache_ptr) {
539 if (!cache_ptr) return;
540 struct cache_s* cache = (struct cache_s*)cache_ptr;
541
542 for (int i = 0; i < CACHE_SHARD_COUNT; i++) {
543 aligned_cache_shard_t* s = &cache->shards[i];
544 fast_rwlock_wrlock(&s->lock);
545 if (s->tags && s->entries) {
546 for (size_t j = 0; j < s->bucket_count; j++) {
547 if (!(s->tags[j] & TAG_CONTROL_MASK)) { entry_ref_dec(s->entries[j]); }
548 }
549 }
550 free(s->tags);
551 free(s->entries);
552 s->tags = NULL;
553 s->entries = NULL;
554 fast_rwlock_unlock_wr(&s->lock);
555 }
556 free(cache);
557}

562const void* cache_get(cache_t* cache_ptr, const char* key, size_t klen, size_t* out_len) {
563 if (unlikely(!cache_ptr || !key || !klen)) return NULL;
564
565 struct cache_s* cache = (struct cache_s*)cache_ptr;
566 uint32_t hash = hash_key(key, klen);
567 uint8_t target_tag = make_tag(hash);
568 aligned_cache_shard_t* shard = &cache->shards[get_shard_idx(hash)];
569
570 fast_rwlock_rdlock(&shard->lock);
571
572 bool found;
573 size_t found_idx = find_slot(shard, hash, key, klen, target_tag, &found);
574 if (!found) {
575 fast_rwlock_unlock_rd(&shard->lock);
576 return NULL;
577 }
578
579 cache_entry_t* entry = shard->entries[found_idx];
580 time_t now = time(NULL);
581
582 if (unlikely(now >= entry->expires_at)) {
583 fast_rwlock_unlock_rd(&shard->lock);
584
585 /* Upgrade to write lock for removal with double-check logic */
586 fast_rwlock_wrlock(&shard->lock);
587 bool re_found;
588 size_t re_idx = find_slot(shard, hash, key, klen, target_tag, &re_found);
589 if (re_found) {
590 cache_entry_t* re_entry = shard->entries[re_idx];
591 if (now >= re_entry->expires_at) {
592 shard->tags[re_idx] = TAG_DELETED;
593 shard->entries[re_idx] = NULL;
594 shard->size--;
595 shard->tombstone_count++;
596 entry_ref_dec(re_entry);
597 }
598 }
599 fast_rwlock_unlock_wr(&shard->lock);
600 return NULL;
601 }
602
603 /* Silent-store optimisation: write the clock bit only if it is clear */
604 if (atomic_load_explicit(&entry->clock_bit, memory_order_relaxed) == 0) {
605 atomic_store_explicit(&entry->clock_bit, 1, memory_order_relaxed);
606 }
607
608 entry_ref_inc(entry);
609
610 if (out_len) *out_len = entry->value_len;
611
612 fast_rwlock_unlock_rd(&shard->lock);
613 return value_ptr_from_entry(entry);
614}

619bool cache_set(cache_t* cache_ptr, const char* key, size_t klen, const void* value, size_t value_len, uint32_t ttl) {
620 if (unlikely(!cache_ptr || !key || !klen || !value || !value_len)) return false;
621
622 struct cache_s* cache = (struct cache_s*)cache_ptr;
623 uint32_t hash = hash_key(key, klen);
624 uint8_t target_tag = make_tag(hash);
625 time_t now = time(NULL);
626
627 /*
628 * Reverted to fast malloc() to avoid aligned_alloc() overhead in glibc.
629 * With hash and key_len moved to offset 0, the hot metadata is guaranteed
630 * to align to a 16-byte boundary and cannot span across a cache-line split.
631 */
632 size_t alloc_sz = offsetof(cache_entry_t, data) + klen + 1 + sizeof(void*) + value_len;
633 cache_entry_t* new_entry = malloc(alloc_sz);
634 if (!new_entry) return false;
635
636 new_entry->hash = hash;
637 new_entry->key_len = (uint32_t)klen;
638 new_entry->value_len = value_len;
639 new_entry->expires_at = now + (time_t)(ttl ? ttl : cache->default_ttl);
640 atomic_init(&new_entry->ref_count, 1);
641 atomic_init(&new_entry->clock_bit, 1);
642
643 unsigned char* dp = new_entry->data;
644 memcpy(dp, key, klen);
645 dp[klen] = '\0';
646 *(cache_entry_t**)(dp + klen + 1) = new_entry;
647 memcpy(dp + klen + 1 + sizeof(cache_entry_t*), value, value_len);
648
649 aligned_cache_shard_t* shard = &cache->shards[get_shard_idx(hash)];
650
651 /* Speculatively prefetch the entry pointer slot before acquiring the lock */
652 size_t mask = shard->bucket_count - 1;
653 size_t idx = hash & mask;
654#if defined(__GNUC__) || defined(__clang__)
655 __builtin_prefetch(&shard->entries[idx], 0, 3);
656#endif
657
658 fast_rwlock_wrlock(&shard->lock);
659
660 if (unlikely(shard->tombstone_count > (shard->bucket_count * 3) / 4)) { compact_shard(shard); }
661
662 bool found;
663 size_t found_idx = find_slot(shard, hash, key, klen, target_tag, &found);
664
665 if (found) {
666 cache_entry_t* old = shard->entries[found_idx];
667 shard->entries[found_idx] = new_entry;
668 entry_ref_dec(old);
669 } else {
670 if (shard->size >= shard->capacity) {
671 if (!clock_evict(shard)) {
672 fast_rwlock_unlock_wr(&shard->lock);
673 free(new_entry);
674 return false;
675 }
676 }
677
678 if (shard->tags[found_idx] == TAG_DELETED) shard->tombstone_count--;
679
680 shard->tags[found_idx] = target_tag;
681 shard->entries[found_idx] = new_entry;
682 shard->size++;
683 }
684
685 fast_rwlock_unlock_wr(&shard->lock);
686 return true;
687}

799bool cache_save(cache_t* cache_ptr, const char* filename) {
800 if (!cache_ptr || !filename) return false;
801 struct cache_s* cache = (struct cache_s*)cache_ptr;
802
803 FILE* f = fopen(filename, "wb");
804 if (!f) return false;
805
806 uint32_t magic = CACHE_FILE_MAGIC;
807 uint32_t version = CACHE_FILE_VERSION;
808 uint64_t total_entries = 0;
809
810 if (!file_write_chk(&magic, sizeof(magic), 1, f) || !file_write_chk(&version, sizeof(version), 1, f) ||
811 !file_write_chk(&total_entries, sizeof(total_entries), 1, f)) {
812 fclose(f);
813 return false;
814 }
815
816 uint64_t actual_count = 0;
817 time_t now = time(NULL);
818
819 for (int i = 0; i < CACHE_SHARD_COUNT; i++) {
820 aligned_cache_shard_t* shard = &cache->shards[i];
821 fast_rwlock_rdlock(&shard->lock);
822
823 for (size_t j = 0; j < shard->bucket_count; j++) {
824 if (shard->tags[j] & TAG_CONTROL_MASK) continue;
825
826 cache_entry_t* entry = shard->entries[j];
827 if (!entry || entry->expires_at <= now) continue;
828
829 uint32_t klen = entry->key_len;
830 uint64_t vlen = (uint64_t)entry->value_len;
831 int64_t expiry = (int64_t)entry->expires_at;
832
833 const void* val_ptr = value_ptr_from_entry(entry);
834 if (!val_ptr) continue;
835
836 if (!file_write_chk(&klen, sizeof(klen), 1, f) || !file_write_chk(&vlen, sizeof(vlen), 1, f) ||
837 !file_write_chk(&expiry, sizeof(expiry), 1, f) || !file_write_chk(entry->data, 1, klen, f) ||
838 !file_write_chk(val_ptr, 1, (size_t)vlen, f)) {
839 fast_rwlock_unlock_rd(&shard->lock);
840 fclose(f);
841 return false;
842 }
843 actual_count++;
844 }
845
846 fast_rwlock_unlock_rd(&shard->lock);
847 }
848
849 fseek(f, (long)(sizeof(magic) + sizeof(version)), SEEK_SET);
850 if (!file_write_chk(&actual_count, sizeof(actual_count), 1, f)) {
851 fclose(f);
852 return false;
853 }
854
855 fclose(f);
856 return true;
857}

862bool cache_load(cache_t* cache_ptr, const char* filename) {

864
865 FILE* f = fopen(filename, "rb");
866 if (!f) return false;

874 fclose(f);
875 return false;
876 }
877
878 if (magic != CACHE_FILE_MAGIC || version != CACHE_FILE_VERSION) {
879 fclose(f);

881 }

883 char key_buf[512];
884 char* big_key_buf = NULL;
885 void* val_buf = NULL;
886 size_t val_buf_cap = 0;
887 time_t now = time(NULL);
888 bool success = true;
889
890 for (uint64_t i = 0; i < stored_count; i++) {
891 uint32_t klen;
892 uint64_t vlen;
893 int64_t expiry;
894
895 if (!file_read_chk(&klen, sizeof(klen), 1, f) || !file_read_chk(&vlen, sizeof(vlen), 1, f) ||
896 !file_read_chk(&expiry, sizeof(expiry), 1, f)) {
897 success = false;
898 break;
899 }
900
901 char* kptr = key_buf;
902 if (klen >= sizeof(key_buf)) {
903 big_key_buf = realloc(big_key_buf, klen + 1);
904 if (!big_key_buf) {
905 success = false;
906 break;
907 }
908 kptr = big_key_buf;
909 }
910
911 if (vlen > val_buf_cap) {
912 void* tmp = realloc(val_buf, vlen);
913 if (!tmp) {
914 success = false;
915 break;
916 }
917 val_buf = tmp;
918 val_buf_cap = vlen;
919 }
920
921 if (!file_read_chk(kptr, 1, klen, f) || !file_read_chk(val_buf, 1, (size_t)vlen, f)) {
922 success = false;
923 break;
924 }
925
926 kptr[klen] = '\0';
927
928 if ((time_t)expiry > now) {
929 uint32_t remaining_ttl = (uint32_t)((time_t)expiry - now);
930 if (!cache_set(cache_ptr, kptr, klen, val_buf, (size_t)vlen, remaining_ttl)) {
931 success = false;
932 break;
933 }
934 }
935 }
936
937 free(big_key_buf);
938 free(val_buf);
939 fclose(f);
940 return success;
941}
