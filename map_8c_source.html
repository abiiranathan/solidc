223// Set a key-value pair with optimizations and better error handling
224bool map_set(HashMap* m, void* key, size_t key_len, void* value) {
225 // Check if we need to resize
226 size_t total_entries = m->size + m->tombstone_count;
227 if (total_entries >= (size_t)((float)m->capacity * m->max_load_factor)) {
228 size_t new_capacity = calculate_new_capacity(m->capacity);
229 if (new_capacity <= m->capacity || !map_resize(m, new_capacity, key_len)) {
230 return false;
231 }
232
233 // Check for tombstone cleanup
234 } else if (m->tombstone_count > 0 && (double)m->tombstone_count / (double)m->size > TOMBSTONE_RATIO_THRESHOLD) {
235 if (!map_resize(m, m->capacity, key_len)) {
236 return false;
237 }
238 }

241 size_t index = hash & (m->capacity - 1); // Bitmask for power-of-two capacity
242 size_t hash2 = (hash >> 5) | 1; // Ensure odd step for probing

244 size_t probe_count = 0;

247 void** current_key = get_key_ptr(m, index);

249 if (*current_key == NULL) {

251 }
252
253 if (m->deleted[index]) {
254 if (first_tombstone == SIZE_MAX) {
255 first_tombstone = index;
256 }
257 } else if (m->key_compare(*current_key, key)) {
258 // Key exists - update value

260 m->value_free(*get_value_ptr(m, index));
261 }
262 *get_value_ptr(m, index) = value;
263 return true;
264 }
265
266 probe_count++;
267 index = (index + hash2) & (m->capacity - 1);
268 }
269
270 // Use first tombstone if we found one
271 if (first_tombstone != SIZE_MAX) {
272 index = first_tombstone;
273 m->tombstone_count--;
274 }
275
276 // Insert new entry
277 *get_key_ptr(m, index) = key;
278 *get_value_ptr(m, index) = value;
279
280 m->deleted[index] = false;
281 m->size++;

283 return true;
