50static inline uint64_t hashset_default_hash(const void* key, size_t key_size) {
51 const uint64_t FNV_OFFSET = 14695981039346656037ULL;
52 const uint64_t FNV_PRIME = 1099511628211ULL;
53
54 uint64_t hash = FNV_OFFSET;
55 const unsigned char* bytes = (const unsigned char*)key;

58 hash ^= bytes[i];
59 hash *= FNV_PRIME;

170 hashset_node_t** new_buckets = (hashset_node_t**)calloc(new_capacity, sizeof(hashset_node_t*));

208 if (set == NULL || key == NULL) {
209 return false;
210 }
211
212 // Check if already exists
