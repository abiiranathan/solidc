28static inline OrthonormalBasis orthonormalize(Vec3 v0, Vec3 v1) {
29 // Load into SIMD registers
30 SimdVec3 sv0 = vec3_load(v0);
31 SimdVec3 sv1 = vec3_load(v1);
32
33 // Normalize v0
34 sv0 = vec3_normalize(sv0);
35
36 // Project v1 onto v0 and subtract to make orthogonal: v1 = v1 - dot(v1, v0) * v0
37 float dot = vec3_dot(sv0, sv1);
38 sv1 = vec3_sub(sv1, vec3_mul(sv0, dot));
39 sv1 = vec3_normalize(sv1);
40
41 // Cross product for v2
42 SimdVec3 sv2 = vec3_cross(sv0, sv1);
43
44 // Store results
45 return (OrthonormalBasis){vec3_store(sv0), vec3_store(sv1), vec3_store(sv2)};
46}

63static inline EigenDecomposition mat3_eigen_symmetric(Mat3 A) {
64 EigenDecomposition result;
65 Mat3 V = mat3_identity();
66
67 const int MAX_ITERS = 32;
68 const float EPSILON = 1e-10f;
69
70 for (int iter = 0; iter < MAX_ITERS; ++iter) {
71 // Find largest off-diagonal absolute value
72 int p = 0, q = 1;
73 float max = fabsf(A.m[0][1]);
74
75 if (fabsf(A.m[0][2]) > max) {
76 p = 0;
77 q = 2;
78 max = fabsf(A.m[0][2]);
79 }
80 if (fabsf(A.m[1][2]) > max) {
81 p = 1;
82 q = 2;
83 max = fabsf(A.m[1][2]);
84 }
85
86 if (max < EPSILON) break;
87
88 float app = A.m[p][p];
89 float aqq = A.m[q][q];
90 float apq = A.m[p][q];
91
92 float phi = 0.5f * atanf((2.0f * apq) / (aqq - app + 1e-20f));
93 float c = cosf(phi);
94 float s = sinf(phi);
95
96 // Perform Jacobi rotation (Scalar)
97 for (int r = 0; r < 3; ++r) {
98 float arp = A.m[r][p];
99 float arq = A.m[r][q];
100 A.m[r][p] = c * arp - s * arq;
101 A.m[r][q] = s * arp + c * arq;
102 }
103
104 for (int r = 0; r < 3; ++r) {
105 float arp = A.m[p][r];
106 float arq = A.m[q][r];
107 A.m[p][r] = c * arp - s * arq;
108 A.m[q][r] = s * arp + c * arq;
109 }
110
111 A.m[p][p] = c * c * app - 2 * s * c * apq + s * s * aqq;
112 A.m[q][q] = s * s * app + 2 * s * c * apq + c * c * aqq;
113 A.m[p][q] = A.m[q][p] = 0.0f;
114
115 // Update eigenvectors
116 for (int r = 0; r < 3; ++r) {
117 float vrp = V.m[r][p];
118 float vrq = V.m[r][q];
119 V.m[r][p] = c * vrp - s * vrq;
120 V.m[r][q] = s * vrp + c * vrq;
121 }
122 }
123
124 result.eigenvalues = (Vec3){A.m[0][0], A.m[1][1], A.m[2][2]};
125 result.eigenvectors = V;
126 return result;
127}

133static inline void mat3_svd(Mat3 A, Mat3* U, Vec3* S, Mat3* V) {
134 // Step 1: Compute ATA = A^T * A
135 // Manual multiplication is fastest for 3x3 here
136 Mat3 ATA = {0};
137 for (int i = 0; i < 3; ++i) { // col of A^T (row of A)
138 for (int j = 0; j < 3; ++j) { // row of A^T (col of A)
139 ATA.m[i][j] = 0.0f;
140 for (int k = 0; k < 3; ++k) {
141 ATA.m[i][j] += A.m[i][k] * A.m[j][k];
142 }
143 }
144 }
145
146 // Step 2: Eigen decomposition of ATA
147 EigenDecomposition ed = mat3_eigen_symmetric(ATA);
148 *V = ed.eigenvectors;
149
150 // Step 3: Sort eigenvalues/vectors (descending)
151 int order[3] = {0, 1, 2};
152 float vals[3] = {ed.eigenvalues.x, ed.eigenvalues.y, ed.eigenvalues.z};
153
154 for (int i = 0; i < 2; ++i) {
155 for (int j = i + 1; j < 3; ++j) {
156 if (vals[order[i]] < vals[order[j]]) {
157 int tmp = order[i];
158 order[i] = order[j];
159 order[j] = tmp;
160 }
161 }
162 }
163
164 Vec3 sorted_eigen = {vals[order[0]], vals[order[1]], vals[order[2]]};
165
166 Mat3 V_sorted = {0};
167 for (int i = 0; i < 3; ++i) {
168 for (int j = 0; j < 3; ++j) {
169 V_sorted.m[i][j] = V->m[order[i]][j];
170 }
171 }
172 *V = V_sorted;
173
174 // Step 4: Compute Singular Values
175 S->x = sqrtf(fmaxf(0.0f, sorted_eigen.x));
176 S->y = sqrtf(fmaxf(0.0f, sorted_eigen.y));
177 S->z = sqrtf(fmaxf(0.0f, sorted_eigen.z));
178
179 // Step 5: Compute U = A * V * S^-1
180 // We treat U columns as SimdVec3 for calculation
181 SimdVec3 colA[3] = {vec3_load((Vec3){A.m[0][0], A.m[0][1], A.m[0][2]}),
182 vec3_load((Vec3){A.m[1][0], A.m[1][1], A.m[1][2]}),
183 vec3_load((Vec3){A.m[2][0], A.m[2][1], A.m[2][2]})};
184
185 float* s_arr = (float*)S;
186 for (int i = 0; i < 3; ++i) { // For each column of U
187 float sigma = s_arr[i];
188 if (sigma > 1e-6f) {
189 // U_col_i = (A * V_col_i) / sigma
190 Vec3 v_col = {V->m[i][0], V->m[i][1], V->m[i][2]};
191 SimdVec3 Av =
192 vec3_add(vec3_add(vec3_mul(colA[0], v_col.x), vec3_mul(colA[1], v_col.y)), vec3_mul(colA[2], v_col.z));
193 SimdVec3 u_col = vec3_mul(Av, 1.0f / sigma);
194
195 // Store back to matrix
196 Vec3 res = vec3_store(u_col);
197 U->m[i][0] = res.x;
198 U->m[i][1] = res.y;
199 U->m[i][2] = res.z;
200 } else {
201 U->m[i][0] = U->m[i][1] = U->m[i][2] = 0.0f;
202 }
203 }
204
205 // Step 6: Orthonormalize U if rank-deficient (using Cross Product)
206 if (S->y > 1e-6f && S->z < 1e-6f) {
207 SimdVec3 u0 = vec3_load((Vec3){U->m[0][0], U->m[0][1], U->m[0][2]});
208 SimdVec3 u1 = vec3_load((Vec3){U->m[1][0], U->m[1][1], U->m[1][2]});
209 SimdVec3 u2 = vec3_cross(u0, u1); // U2 = U0 x U1
210
211 Vec3 res = vec3_store(u2);
212 U->m[2][0] = res.x;
213 U->m[2][1] = res.y;
214 U->m[2][2] = res.z;
215 }
216
217 // Step 7: Fix det(U) = +1
218 if (mat3_determinant(*U) < 0.0f) {
219 U->m[2][0] = -U->m[2][0];
220 U->m[2][1] = -U->m[2][1];
221 U->m[2][2] = -U->m[2][2];
222 }
223}

229static inline void mat4_qr(Mat4 A, Mat4* Q, Mat4* R) {
230 // Load columns into SIMD registers
231 SimdVec4 v[4], q[4];
232 for (int i = 0; i < 4; ++i) {
233 v[i].v = A.cols[i]; // Direct access to underlying SIMD type
234 }
235
236 for (int i = 0; i < 4; ++i) {
237 q[i] = v[i];
238 for (int j = 0; j < i; ++j) {
239 float r = vec4_dot(q[j], v[i]);
240 R->m[j][i] = r;
241
242 // q[i] = q[i] - r * q[j]
243 q[i] = vec4_sub(q[i], vec4_mul(q[j], r));
244 }
245
246 float norm = vec4_length(q[i]);
247 if (norm < 1e-6f) norm = 1e-6f;
248
249 R->m[i][i] = norm;
250 q[i] = vec4_mul(q[i], 1.0f / norm);
251 }
252
253 // Store Q columns
254 for (int i = 0; i < 4; ++i) {
255 Q->cols[i] = q[i].v;
256 }
257
258 // Zero lower triangle of R
259 for (int i = 0; i < 4; ++i)
260 for (int j = 0; j < i; ++j)
261 R->m[i][j] = 0.0f;
262}

267static inline void mat4_power_iteration(Mat4 A, Vec4* eigenvector, float* eigenvalue, int max_iter, float tol) {
268 SimdVec4 v = vec4_load((Vec4){1.0f, 0.0f, 0.0f, 0.0f});
269 SimdVec4 Av;
270 float lambda_old = 0.0f;
271
272 for (int iter = 0; iter < max_iter; ++iter) {
273 // Av = A * v (SIMD matrix-vector mul)
274 // We cast A to Mat4 to use the function from matrix.h
275 Vec4 v_scalar = vec4_store(v);
276 Vec4 Av_scalar = mat4_mul_vec4(A, v_scalar);
277 Av = vec4_load(Av_scalar);
278
279 // lambda = dot(Av, v)
280 *eigenvalue = vec4_dot(Av, v);
281
282 // Normalize
283 Av = vec4_normalize(Av);
284
285 if (fabsf(*eigenvalue - lambda_old) < tol) {
286 break;
287 }
288 lambda_old = *eigenvalue;
289 v = Av;
290 }
291
292 *eigenvector = vec4_store(v);
293}

298static inline float mat4_norm_frobenius(Mat4 A) {
299 // Sum of squares of all elements = Sum of dot product of each column with itself
300 SimdVec4 c0 = {.v = A.cols[0]};
301 SimdVec4 c1 = {.v = A.cols[1]};
302 SimdVec4 c2 = {.v = A.cols[2]};
303 SimdVec4 c3 = {.v = A.cols[3]};
304
305 float sum = 0.0f;
306 sum += vec4_dot(c0, c0);
307 sum += vec4_dot(c1, c1);
308 sum += vec4_dot(c2, c2);
309 sum += vec4_dot(c3, c3);
310
311 return sqrtf(sum);
312}

314static inline bool mat3_is_positive_definite(Mat3 A) {
315 // 1. Leading principal minor 1: A[0][0]
316 if (A.m[0][0] <= 0.0f) return false;
317
318 // 2. Leading principal minor 2: det 2x2
319 float det2 = A.m[0][0] * A.m[1][1] - A.m[1][0] * A.m[0][1];
320 if (det2 <= 0.0f) return false;
321
322 // 3. Leading principal minor 3: det 3x3
323 if (mat3_determinant(A) <= 0.0f) return false;
324
325 return true;
326}

328static inline float mat4_condition_number(Mat4 A) {
329 float norm_A = mat4_norm_frobenius(A);
330 Mat4 A_inv = mat4_inverse(A);
331
332 if (mat4_equal(A_inv, mat4_identity())) {
333 return FLT_MAX; // Singular
334 }
335
336 float norm_A_inv = mat4_norm_frobenius(A_inv);
337 return norm_A * norm_A_inv;
338}

342static inline bool mat4_lu(Mat4 A, Mat4* L, Mat4* U, Mat4* P) {
343 const float tolerance = 1e-6f;
344 *U = A;
345
346 // Identity L and P
347 *L = mat4_identity();
348 *P = mat4_identity();
349
350 for (int k = 0; k < 4; ++k) {
351 // Pivot
352 int pivot_row = k;
353 float max_val = fabsf(U->m[k][k]);
354 for (int i = k + 1; i < 4; ++i) {
355 float val = fabsf(U->m[k][i]);
356 if (val > max_val) {
357 max_val = val;
358 pivot_row = i;
359 }
360 }
361
362 if (max_val < tolerance) return false;
363
364 // Swap Rows
365 if (pivot_row != k) {
366 for (int j = 0; j < 4; ++j) {
367 float tmp;
368 tmp = U->m[j][k];
369 U->m[j][k] = U->m[j][pivot_row];
370 U->m[j][pivot_row] = tmp;
371 tmp = P->m[j][k];
372 P->m[j][k] = P->m[j][pivot_row];
373 P->m[j][pivot_row] = tmp;
374 if (j < k) {
375 tmp = L->m[j][k];
376 L->m[j][k] = L->m[j][pivot_row];
377 L->m[j][pivot_row] = tmp;
378 }
379 }
380 }
381
382 // Eliminate
383 for (int i = k + 1; i < 4; ++i) {
384 float factor = U->m[k][i] / U->m[k][k];
385 L->m[k][i] = factor;
386 for (int j = k; j < 4; ++j) {
387 U->m[j][i] -= factor * U->m[j][k];
388 }
389 }
390 }
391 return true;
392}

394static inline Vec4 forward_substitution_mat4(Mat4 L, Vec4 b) {
395 Vec4 x;
396 float* x_arr = &x.x;
397 x_arr[0] = b.x / L.m[0][0];
398 x_arr[1] = (b.y - L.m[0][1] * x_arr[0]) / L.m[1][1];
399 x_arr[2] = (b.z - (L.m[0][2] * x_arr[0] + L.m[1][2] * x_arr[1])) / L.m[2][2];
400 x_arr[3] = (b.w - (L.m[0][3] * x_arr[0] + L.m[1][3] * x_arr[1] + L.m[2][3] * x_arr[2])) / L.m[3][3];
401 return x;
402}

404static inline Vec4 backward_substitution_mat4(Mat4 U, Vec4 b) {
405 Vec4 x;
406 float* x_arr = &x.x;
407 x_arr[3] = b.w / U.m[3][3];
408 x_arr[2] = (b.z - U.m[3][2] * x_arr[3]) / U.m[2][2];
409 x_arr[1] = (b.y - (U.m[2][1] * x_arr[2] + U.m[3][1] * x_arr[3])) / U.m[1][1];
410 x_arr[0] = (b.x - (U.m[1][0] * x_arr[1] + U.m[2][0] * x_arr[2] + U.m[3][0] * x_arr[3])) / U.m[0][0];
411 return x;
412}

414static inline Vec4 mat4_solve(Mat4 A, Vec4 b) {
415 Mat4 L, U, P;
416 if (!mat4_lu(A, &L, &U, &P)) {
417 return (Vec4){0};
418 }
419
420 // Permute b: Pb
421 Vec4 Pb;
422 Pb.x = P.m[0][0] * b.x + P.m[1][0] * b.y + P.m[2][0] * b.z + P.m[3][0] * b.w;
423 Pb.y = P.m[0][1] * b.x + P.m[1][1] * b.y + P.m[2][1] * b.z + P.m[3][1] * b.w;
424 Pb.z = P.m[0][2] * b.x + P.m[1][2] * b.y + P.m[2][2] * b.z + P.m[3][2] * b.w;
425 Pb.w = P.m[0][3] * b.x + P.m[1][3] * b.y + P.m[2][3] * b.z + P.m[3][3] * b.w;
426
427 Vec4 y = forward_substitution_mat4(L, Pb);
428 return backward_substitution_mat4(U, y);
429}

431static inline Vec3 mat3_solve(Mat3 A, Vec3 b) {
432 Mat3 L, U, P;
433 if (!mat3_lu(A, &L, &U, &P)) return (Vec3){0};
434
435 Vec3 Pb;
436 Pb.x = P.m[0][0] * b.x + P.m[1][0] * b.y + P.m[2][0] * b.z;
437 Pb.y = P.m[0][1] * b.x + P.m[1][1] * b.y + P.m[2][1] * b.z;
438 Pb.z = P.m[0][2] * b.x + P.m[1][2] * b.y + P.m[2][2] * b.z;
439
440 Vec3 y = forward_substitution_mat3(L, Pb);
441 return backward_substitution_mat3(U, y);
442}
