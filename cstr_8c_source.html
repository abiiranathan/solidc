161static size_t calculate_growth_capacity(size_t current_cap, size_t min_needed) {
162 if (min_needed > CSTR_MAX_SIZE) return 0;
163
164 min_needed = MAX(min_needed, CSTR_MIN_HEAP_CAPACITY); // ensure min heap capacity
165 current_cap = MAX(current_cap, CSTR_MIN_HEAP_CAPACITY); // avoid multiplying by 0.
166
167 size_t capacity = 0;
168 bool overflow = false;
169
170 // grow capacity as long as we haven't overflow'd and still below required memory.
171 while (!(overflow = ckd_mul(&capacity, current_cap, CSTR_GROWTH_FACTOR)) && capacity < min_needed) {
172 current_cap = capacity; // update current capacity.
173 };
174
175 if (overflow) return 0;
176 return (overflow || capacity > CSTR_MAX_SIZE) ? 0 : capacity;
177}

192static bool cstr_promote_to_heap(cstr* s, size_t new_capacity) {
193 // Calculate actual allocation size (powers of 2, etc.)
194 size_t cap = calculate_growth_capacity(SSO_MAX_SIZE, new_capacity);
195 if (cap == 0) return false;
196
197 char* new_mem = NULL;
198 if (s->arena) {
199 new_mem = arena_alloc(s->arena, cap);
200 } else {
201 new_mem = malloc(cap);
202 }
203
204 if (!new_mem) return false;
205
206 // COPY data from Stack to the new Heap buffer
207 // We must do this BEFORE modifying the struct, because writing to
208 // s->heap members might overwrite s->stack.data members!
209 memcpy(new_mem, s->stack.data, s->stack.len);
210
211 // Null terminate the new buffer
212 new_mem[s->stack.len] = '\0';
213 size_t old_len = s->stack.len;
214 s->heap.data = new_mem;
215 s->heap.length = old_len;
216 s->heap.capacity = CSTR_SET_HEAP_CAPACITY(cap);
217
218 return true;
219}

233static bool cstr_ensure_capacity(cstr* s, size_t new_capacity) {
234 if (new_capacity <= cstr_get_capacity(s)) {
235 return true;
236 }
237
238 // Prevent overflow attacks
239 if (new_capacity > CSTR_MAX_SIZE) {
240 return false;
241 }
242
243 if (CSTR_IS_HEAP(s)) {
244 size_t optimal_capacity = calculate_growth_capacity(cstr_get_capacity(s), new_capacity);
245 if (optimal_capacity == 0) {
246 return false;
247 }
248
249 char* new_data = NULL;
250 if (s->arena) {
251 // Arena doesn't support realloc, so alloc new + copy
252 new_data = arena_alloc(s->arena, optimal_capacity);
253 if (new_data) {
254 memcpy(new_data, s->heap.data, s->heap.length + 1);
255 // Old data is left in arena (leak within arena scope)
256 }
257 } else {
258 new_data = realloc(s->heap.data, optimal_capacity);
259 }
260
261 if (!new_data) {
262 return false;
263 }
264
265 s->heap.data = new_data;
266 s->heap.capacity = CSTR_SET_HEAP_CAPACITY(optimal_capacity);
267 } else {
268 return cstr_promote_to_heap(s, new_capacity);
269 }
270
271 return true;
272}

297cstr* cstr_init_arena(Arena* arena, size_t initial_capacity) {
298 if (initial_capacity > CSTR_MAX_SIZE) return NULL;
299
300 cstr* s = NULL;
301 if (arena) {
302 s = ARENA_ALLOC(arena, cstr);
303 } else {
304 s = malloc(sizeof(cstr));
305 }
306
307 if (!s) return NULL;
308
309 s->arena = arena;
310 initial_capacity = MAX(initial_capacity, STR_MIN_CAPACITY);
311
312 if (initial_capacity <= SSO_MAX_SIZE) {
313 // Initialize as stack-allocated string
314 s->stack.len = 0;
315 s->stack.data[0] = '\0';
316 // Clear remaining bytes for security
317 memset(s->stack.data + 1, 0, SSO_MAX_SIZE - 1);
318 } else {
319 size_t optimal_capacity = calculate_growth_capacity(0, initial_capacity);
320 if (optimal_capacity == 0) {
321 if (!arena) free(s);
322 return NULL;
323 }
324
325 char* data = NULL;
326 if (arena) {
327 data = arena_alloc(arena, optimal_capacity);
328 } else {
329 data = ALIGNED_ALLOC(STR_ALIGNMENT, optimal_capacity);
330 }
331
332 if (!data) {
333 if (!arena) free(s);
334 return NULL;
335 }
336
337 data[0] = '\0';
338 s->heap.data = data;
339 s->heap.length = 0;
340 s->heap.capacity = CSTR_SET_HEAP_CAPACITY(optimal_capacity);
341 }
342
343 return s;
344}

870size_t cstr_remove_all(cstr* s, const char* substr) {
871 if (!s || !substr || !*substr) {
872 return 0;
873 }
874
875 size_t substr_len = strlen(substr);
876 if (substr_len == 0) {
877 return 0; // Cannot remove an empty substring
878 }
879
880 char* data = cstr_get_data(s);
881 char* write_ptr = data;
882 char* read_ptr = data;
883 size_t count = 0;
884 const char* end = data + cstr_get_length(s);
885
886 while (read_ptr < end) {
887 // Check if the substring exists starting from read_ptr
888 if ((size_t)(end - read_ptr) >= substr_len && strncmp(read_ptr, substr, substr_len) == 0) {
889 read_ptr += substr_len; // Skip the substring
890 count++;
891 } else {
892 *write_ptr++ = *read_ptr++; // Copy the character
893 }
894 }
895 *write_ptr = '\0'; // Null-terminate the resulting string
896 cstr_set_length(s, (size_t)(write_ptr - data));
897 return count;

1681 count++;

1690 size_t result_len_estimate = 0;

1696 } else {
1697 size_t delta = 0;

1700 }

1702 }

1722 memcpy(dest + pos, new_sub, new_len);

1729 return NULL;

1756cstr** cstr_split(const cstr* s, const char* delim, size_t* count_out) {
1757 if (!s || !count_out) {
1758 if (count_out) *count_out = 0;
1759 return NULL;
1760 }
1761
1762 const char* s_data = cstr_get_data_const(s);
1763 size_t s_len = cstr_get_length(s);
1764 *count_out = 0;
1765
1766 // Handle empty delimiter case
1767 if (!delim || !*delim) {
1768 cstr** result = malloc(sizeof(cstr*));
1769 if (!result) return NULL;
1770
1771 result[0] = cstr_new_arena(s->arena, s_data);
1772 if (!result[0]) {
1773 free(result);
1774 return NULL;
1775 }
1776 *count_out = 1;
1777 return result;
1778 }
1779
1780 size_t delim_len = strlen(delim);
1781
1782 // Estimate capacity (avoid scanning entire string)
1783 size_t capacity = (s_len / (delim_len * 2)) + 4;
1784 cstr** result = malloc(capacity * sizeof(cstr*));
1785 if (!result) return NULL;
1786
1787 const char* start = s_data;
1788 const char* end = s_data + s_len;
1789 size_t count = 0;
1790
1791 while (1) {
1792 const char* found = strstr(start, delim);
1793 if (!found || found >= end) found = end;
1794
1795 // Grow array if needed
1796 if (count >= capacity) {
1797 capacity *= 2;
1798 cstr** new_result = realloc(result, capacity * sizeof(cstr*));
1799 if (!new_result) goto error;
1800 result = new_result;
1801 }
1802
1803 size_t token_len = (size_t)(found - start);
1804 // Use s->arena for substrings
1805 result[count] = cstr_init_arena(s->arena, token_len + 1);
1806 if (!result[count]) goto error;
1807
1808 char* data = cstr_get_data(result[count]);
1809 memcpy(data, start, token_len);
1810 data[token_len] = '\0';
1811 cstr_set_length(result[count], token_len);
1812
1813 count++;
1814
1815 if (found == end) break;
1816 start = found + delim_len;
1817 }
1818
1819 // Trim to exact size if we over-allocated significantly
1820 if (capacity > count * 2 && count > 8) {
1821 cstr** trimmed = realloc(result, count * sizeof(cstr*));
1822 if (trimmed) result = trimmed;
1823 }
1824
1825 *count_out = count;
1826 return result;
1827
1828error:
1829 for (size_t i = 0; i < count; i++) {
1830 cstr_free(result[i]);
1831 }
1832 free(result);
1833 *count_out = 0;
1834 return NULL;
1835}

1863 // Invalid input array
1864 if (!strings[i]) {
1865 return NULL;
1866 }
1867
