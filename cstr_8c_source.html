160static size_t calculate_growth_capacity(size_t current_cap, size_t min_needed) {
161 if (min_needed > CSTR_MAX_SIZE) return 0;
162
163 min_needed = MAX(min_needed, CSTR_MIN_HEAP_CAPACITY); // ensure min heap capacity
164 current_cap = MAX(current_cap, CSTR_MIN_HEAP_CAPACITY); // avoid multiplying by 0.
165
166 size_t capacity = 0;
167 bool overflow = false;
168
169 // grow capacity as long as we haven't overflow'd and still below required memory.
170 while (!(overflow = ckd_mul(&capacity, current_cap, CSTR_GROWTH_FACTOR)) && capacity < min_needed) {
171 current_cap = capacity; // update current capacity.
172 };
173
174 if (overflow) return 0;
175 return (overflow || capacity > CSTR_MAX_SIZE) ? 0 : capacity;
176}

191static bool cstr_promote_to_heap(cstr* s, size_t new_capacity) {
192 // Calculate actual allocation size (powers of 2, etc.)
193 size_t cap = calculate_growth_capacity(SSO_MAX_SIZE, new_capacity);
194 if (cap == 0) return false;
195
196 // Allocate the NEW memory first
197 char* new_mem = malloc(cap);
198 if (!new_mem) return false;
199
200 // COPY data from Stack to the new Heap buffer
201 // We must do this BEFORE modifying the struct, because writing to
202 // s->heap members might overwrite s->stack.data members!
203 memcpy(new_mem, s->stack.data, s->stack.len);
204
205 // Null terminate the new buffer
206 new_mem[s->stack.len] = '\0';
207 size_t old_len = s->stack.len;
208 s->heap.data = new_mem;
209 s->heap.length = old_len;
210 s->heap.capacity = CSTR_SET_HEAP_CAPACITY(cap);
211
212 return true;
213}

227static bool cstr_ensure_capacity(cstr* s, size_t new_capacity) {
228 if (new_capacity <= cstr_get_capacity(s)) {
229 return true;
230 }
231
232 // Prevent overflow attacks
233 if (new_capacity > CSTR_MAX_SIZE) {
234 return false;
235 }
236
237 if (CSTR_IS_HEAP(s)) {
238 size_t optimal_capacity = calculate_growth_capacity(cstr_get_capacity(s), new_capacity);
239 if (optimal_capacity == 0) {
240 return false;
241 }
242
243 char* new_data = realloc(s->heap.data, optimal_capacity);
244 if (!new_data) {
245 return false;
246 }
247
248 s->heap.data = new_data;
249 s->heap.capacity = CSTR_SET_HEAP_CAPACITY(optimal_capacity);
250 } else {
251 return cstr_promote_to_heap(s, new_capacity);
252 }
253
254 return true;
255}

271cstr* cstr_init(size_t initial_capacity) {
272 if (initial_capacity > CSTR_MAX_SIZE) return NULL;
273
274 cstr* s = malloc(sizeof(cstr));
275 if (!s) return NULL;
276
277 initial_capacity = MAX(initial_capacity, STR_MIN_CAPACITY);
278
279 if (initial_capacity <= SSO_MAX_SIZE) {
280 // Initialize as stack-allocated string
281 s->stack.len = 0;
282 s->stack.data[0] = '\0';
283 // Clear remaining bytes for security
284 memset(s->stack.data + 1, 0, SSO_MAX_SIZE - 1);
285 } else {
286 size_t optimal_capacity = calculate_growth_capacity(0, initial_capacity);
287 if (optimal_capacity == 0) {
288 free(s);
289 return NULL;
290 }
291
292 char* data = ALIGNED_ALLOC(STR_ALIGNMENT, optimal_capacity);
293 if (!data) {
294 free(s);
295 return NULL;
296 }
297
298 data[0] = '\0';
299 s->heap.data = data;
300 s->heap.length = 0;
301 s->heap.capacity = CSTR_SET_HEAP_CAPACITY(optimal_capacity);
302 }
303
304 return s;
305}

545cstr* cstr_format(const char* format, ...) {
546 if (!format) {
547 return NULL;
548 }
549
550 va_list args;
551 va_start(args, format);
552
553 // Get required length
554 va_list args_copy;
555 va_copy(args_copy, args);
556 int len = vsnprintf(NULL, 0, format, args_copy);
557 va_end(args_copy);
558
559 if (len < 0) {
560 va_end(args);
561 return NULL;
562 }
563
564 // Prevent unreasonably large format results
565 if ((size_t)len >= CSTR_MAX_SIZE) {
566 va_end(args);
567 return NULL;
568 }
569
570 size_t capacity = (size_t)len + 1;
571 cstr* s = cstr_init(capacity);
572 if (!s) {
573 va_end(args);
574 return NULL;
575 }
576
577 char* data = cstr_get_data(s);
578 int written = vsnprintf(data, capacity, format, args);
579 va_end(args);
580
581 if (written < 0 || (size_t)written >= capacity) {
582 cstr_free(s);
583 return NULL;
584 }
585
586 cstr_set_length(s, (size_t)written);
587 return s;
588}

776bool cstr_remove(cstr* s, size_t index, size_t count) {
777 if (!s) {
778 return false;
779 }
780 size_t current_length = cstr_get_length(s);
781
782 if (index > current_length) {
783 // Invalid index, unless count is 0 and index is exactly the length
784 return index == current_length && count == 0;
785 }
786
787 // Adjust count if it exceeds the remaining length
788 count = MIN(count, current_length - index);
789
790 if (count == 0) {
791 return true; // No characters to remove
792 }
793
794 char* data = cstr_get_data(s);
795 // Shift the remaining part of the string
796 memmove(data + index, data + index + count,
797 current_length - index - count + 1); // +1 for the null terminator
798 cstr_set_length(s, current_length - count);
799 return true;
800}

827size_t cstr_remove_all(cstr* s, const char* substr) {
828 if (!s || !substr || !*substr) {
829 return 0;
830 }
831
832 size_t substr_len = strlen(substr);
833 if (substr_len == 0) {
834 return 0; // Cannot remove an empty substring
835 }
836
837 char* data = cstr_get_data(s);
838 char* write_ptr = data;
839 char* read_ptr = data;
840 size_t count = 0;
841 const char* end = data + cstr_get_length(s);
842
843 while (read_ptr < end) {
844 // Check if the substring exists starting from read_ptr
845 if ((size_t)(end - read_ptr) >= substr_len && strncmp(read_ptr, substr, substr_len) == 0) {
846 read_ptr += substr_len; // Skip the substring
847 count++;
848 } else {
849 *write_ptr++ = *read_ptr++; // Copy the character
850 }
851 }
852 *write_ptr = '\0'; // Null-terminate the resulting string
853 cstr_set_length(s, (size_t)(write_ptr - data));
854 return count;
855}

1099bool cstr_snakecase(cstr* s) {
1100 if (!s) return false;
1101
1102 size_t original_length = cstr_get_length(s);
1103 if (original_length == 0) {
1104 return true; // Empty string is already snake_case
1105 }
1106
1107 size_t underscores = 0;
1108 const char* data_const = cstr_get_data_const(s);
1109
1110 // First pass to count needed underscores
1111 for (size_t i = 0; i < original_length; ++i) {
1112 if (i > 0 && isupper((unsigned char)data_const[i])) {
1113 underscores++;
1114 }
1115 }
1116
1117 size_t new_length = original_length + underscores;
1118 if (!cstr_ensure_capacity(s, new_length + 1)) {
1119 return false; // Allocation failure
1120 }
1121
1122 char* data = cstr_get_data(s);
1123 size_t read_idx = original_length;
1124 size_t write_idx = new_length;
1125
1126 // Start from the end to insert underscores without moving data multiple times
1127 while (read_idx > 0) {
1128 read_idx--;

1130

1719cstr** cstr_split(const cstr* s, const char* delim, size_t* count_out) {
1720 if (!s || !count_out) {
1721 if (count_out) *count_out = 0;
1722 return NULL;
1723 }
1724
1725 const char* s_data = cstr_get_data_const(s);
1726 size_t s_len = cstr_get_length(s);
1727 *count_out = 0;
1728
1729 // Handle empty delimiter case
1730 if (!delim || !*delim) {
1731 cstr** result = malloc(sizeof(cstr*));
1732 if (!result) return NULL;
1733 result[0] = cstr_new(s_data);
1734 if (!result[0]) {
1735 free(result);
1736 return NULL;
1737 }
1738 *count_out = 1;
1739 return result;
1740 }
1741
1742 size_t delim_len = strlen(delim);
1743 if (delim_len == 0) {
1744 return cstr_split(s, NULL, count_out);
1745 }
1746
1747 // Estimate capacity (avoid scanning entire string)
1748 size_t capacity = (s_len / (delim_len * 2)) + 4; // Heuristic
1749 cstr** result = malloc(capacity * sizeof(cstr*));
1750 if (!result) return NULL;
1751
1752 const char* start = s_data;
1753 const char* end = s_data + s_len;
1754 size_t count = 0;
1755
1756 while (1) {
1757 const char* found = strstr(start, delim);
1758 if (!found || found >= end) found = end;
1759
1760 // Grow array if needed (but less frequently)
1761 if (count >= capacity) {
1762 capacity *= 2;
1763 cstr** new_result = realloc(result, capacity * sizeof(cstr*));
1764 if (!new_result) goto error;
1765 result = new_result;
1766 }
1767
1768 size_t token_len = (size_t)(found - start);
1769 result[count] = cstr_init(token_len + 1);
1770 if (!result[count]) goto error;
1771
1772 memcpy(cstr_get_data(result[count]), start, token_len);
1773 cstr_get_data(result[count])[token_len] = '\0';
1774 cstr_set_length(result[count], token_len);
1775 count++;
1776
1777 if (found == end) break;
1778 start = found + delim_len;
1779 }
1780
1781 // Trim to exact size if we over-allocated significantly
1782 if (capacity > count * 2 && count > 8) {
1783 cstr** trimmed = realloc(result, count * sizeof(cstr*));
1784 if (trimmed) result = trimmed; // Accept if successful, keep old if not
1785 }
1786
1787 *count_out = count;
1788 return result;
1789
1790error:
1791 for (size_t i = 0; i < count; i++) {
1792 cstr_free(result[i]);
1793 }
1794 free(result);
1795 *count_out = 0;
1796 return NULL;
1797}
