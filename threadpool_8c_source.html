86static bool ringbuffer_push(TaskRingBuffer* rb, Task* task) {
87 // Fast path: try lock-free push
88 uint32_t head = atomic_load_explicit(&rb->head, memory_order_acquire);
89 uint32_t next_head = (head + 1) & RING_BUFFER_MASK;
90 uint32_t tail = atomic_load_explicit(&rb->tail, memory_order_acquire);
91
92 if (next_head != tail) {
93 rb->tasks[head] = *task;
94 atomic_store_explicit(&rb->head, next_head, memory_order_release);
95
96 // Only signal if consumers are actually waiting
97 if (atomic_load_explicit(&rb->waiting_consumers, memory_order_acquire) > 0) {
98 lock_acquire(&rb->mutex);
99 cond_signal(&rb->not_empty);
100 lock_release(&rb->mutex);
101 }
102 return true;
103 }
104
105 // Slow path: blocking push (queue full)
106 lock_acquire(&rb->mutex);
107
108 // Re-check condition under lock
109 while (((atomic_load_explicit(&rb->head, memory_order_relaxed) + 1) & RING_BUFFER_MASK) ==
110 atomic_load_explicit(&rb->tail, memory_order_relaxed)) {
111 if (atomic_load_explicit(&rb->pool->shutdown, memory_order_acquire)) {
112 lock_release(&rb->mutex);
113 return false;
114 }
115 cond_wait(&rb->not_full, &rb->mutex);
116 }
117
118 head = atomic_load_explicit(&rb->head, memory_order_relaxed);
119 rb->tasks[head] = *task;
120 atomic_store_explicit(&rb->head, (head + 1) & RING_BUFFER_MASK, memory_order_release);
121 cond_signal(&rb->not_empty);
122 lock_release(&rb->mutex);
123 return true;
124}

126static int ringbuffer_pull_batch_optimized(TaskRingBuffer* rb, Task* tasks, size_t max_tasks) {
127 int yield_count = 0;
128
129 while (1) {
130 // Single memory barrier for both loads
131 uint32_t tail = atomic_load_explicit(&rb->tail, memory_order_acquire);
132 uint32_t head = atomic_load_explicit(&rb->head, memory_order_relaxed); // relaxed since tail has acquire
133
134 if (head != tail) {
135 // Calculate available tasks
136 size_t available = (head - tail) & RING_BUFFER_MASK;
137 size_t to_take = (available < max_tasks) ? available : max_tasks;
138
139 // Atomic reservation with single CAS
140 uint32_t new_tail = (tail + to_take) & RING_BUFFER_MASK;
141 if (atomic_compare_exchange_weak_explicit(&rb->tail, &tail, new_tail, memory_order_release,
142 memory_order_relaxed)) {
143 // Bulk copy tasks (optimized for cache)
144 for (size_t i = 0; i < to_take; i++) {
145 tasks[i] = rb->tasks[(tail + i) & RING_BUFFER_MASK];
146 }
147
148 // Signal producers only if buffer was full
149 if (available == RING_BUFFER_SIZE - 1) {
150 lock_acquire(&rb->mutex);
151 cond_signal(&rb->not_full);
152 lock_release(&rb->mutex);
153 }
154
155 return (int)to_take;
156 }
157 // CAS failed, retry without delay
158 continue;
159 }
160
161 // Check shutdown (single load)
162 if (atomic_load_explicit(&rb->pool->shutdown, memory_order_acquire)) {
163 return -1;
164 }
165
166 // Reduced yielding to minimize cache pollution
167 if (yield_count < YIELD_THRESHOLD) {
168 yield_count++;
169 thread_yield();
170 continue;
171 }
172
173 // Block efficiently
174 atomic_fetch_add_explicit(&rb->waiting_consumers, 1, memory_order_relaxed);
175
176 lock_acquire(&rb->mutex);
177 // Single check under lock
178 if (atomic_load_explicit(&rb->head, memory_order_relaxed) ==
179 atomic_load_explicit(&rb->tail, memory_order_relaxed) &&
180 !atomic_load_explicit(&rb->pool->shutdown, memory_order_relaxed)) {
181 cond_wait(&rb->not_empty, &rb->mutex);
182 }
183 lock_release(&rb->mutex);
184
185 atomic_fetch_sub_explicit(&rb->waiting_consumers, 1, memory_order_relaxed);
186 yield_count = 0;
187 }
188}

198static inline int has_work(Threadpool* pool) {
199 // Check global queue first (most common case)
200 if (atomic_load_explicit(&pool->queue.head, memory_order_relaxed) !=
201 atomic_load_explicit(&pool->queue.tail, memory_order_relaxed)) {
202 return 1;
203 }
204
205 // Check thread queues
206 for (size_t i = 0; i < pool->num_threads; i++) {
207 if (atomic_load_explicit(&pool->threads[i]->queue.head, memory_order_relaxed) !=
208 atomic_load_explicit(&pool->threads[i]->queue.tail, memory_order_relaxed)) {
209 return 1;
210 }
211 }
212 return 0;
213}

216static void* thread_worker(void* arg) {
217 thread* thp = (thread*)arg;
218 Threadpool* pool = thp->pool;
219 Task batch[BATCH_SIZE] = {0};
220 int batch_count = 0;
221
222 atomic_fetch_add_explicit(&pool->num_threads_alive, 1, memory_order_relaxed);
223
224 while (!atomic_load_explicit(&pool->shutdown, memory_order_acquire)) {
225 // Try local queue first (better cache locality)
226 batch_count = ringbuffer_pull_batch_optimized(&thp->queue, batch, BATCH_SIZE);
227 if (batch_count > 0) {
228 goto execute_batch;
229 }
230
231 // Try global queue
232 batch_count = ringbuffer_pull_batch_optimized(&pool->queue, batch, BATCH_SIZE);
233 if (batch_count > 0) {
234 goto execute_batch;
235 }
236
237 // No work found - continue loop
238 continue;
239
240 execute_batch:
241 // Execute batch without tracking
242 for (int i = 0; i < batch_count; i++) {
243 batch[i].function(batch[i].arg);
244 }
245
246 // Signal idle condition only after completing work
247 if (!has_work(pool)) {
248 lock_acquire(&pool->thcount_lock);
249 // Double-check under lock
250 if (!has_work(pool)) {
251 cond_broadcast(&pool->threads_all_idle);
252 }
253 lock_release(&pool->thcount_lock);
254 }
255 }
256
257 // Thread cleanup
258 int alive = atomic_fetch_sub_explicit(&pool->num_threads_alive, 1, memory_order_acq_rel) - 1;
259 if (alive == 0) {
260 lock_acquire(&pool->thcount_lock);
261 cond_broadcast(&pool->threads_all_idle);
262 lock_release(&pool->thcount_lock);
263 }
264
265 return NULL;
266}

282Threadpool* threadpool_create(size_t num_threads) {
283 if (num_threads == 0) num_threads = 1;
284
285 Threadpool* pool = (Threadpool*)ALIGNED_ALLOC(CACHE_LINE_SIZE, sizeof(Threadpool));
286 if (pool == NULL) return NULL;
287
288 // Initialize minimal atomics
289 atomic_store_explicit(&pool->num_threads_alive, 0, memory_order_relaxed);
290 atomic_store_explicit(&pool->shutdown, 0, memory_order_relaxed);
291 ringbuffer_init(&pool->queue);
292 pool->num_threads = num_threads;
293 pool->queue.pool = pool;
294
295 pool->threads = (thread**)malloc(num_threads * sizeof(thread*));
296 if (pool->threads == NULL) {
297 ringbuffer_destroy(&pool->queue);
298 free(pool);
299 return NULL;
300 }
301
302 lock_init(&pool->thcount_lock);
303 cond_init(&pool->threads_all_idle);
304
305 // Create worker threads
306 for (size_t i = 0; i < num_threads; i++) {
307 if (thread_init(pool, &pool->threads[i]) != 0) {
308 threadpool_destroy(pool, -1);
309 return NULL;
310 }
311 }
312 return pool;

316bool threadpool_submit(Threadpool* pool, void (*function)(void*), void* arg) {
317 if (pool == NULL || function == NULL) return -1;
318
319 Task task = {function, arg};
320
321 // Simple round-robin (no atomic counter)
322 static _Thread_local size_t thread_counter = 0;
323 size_t thread_id = (thread_counter++) % pool->num_threads;
324
325 // Try local queue first
326 if (ringbuffer_push(&pool->threads[thread_id]->queue, &task)) {
327 return true;
328 }
329
330 // Fallback to global queue
331 return ringbuffer_push(&pool->queue, &task);
332}

335void threadpool_destroy(Threadpool* pool, int timeout_ms) {
336 if (pool == NULL) return;
337
338 // Wait for work completion
339 lock_acquire(&pool->thcount_lock);
340
341 int ret = 0;
342 while (has_work(pool)) {
343 ret = cond_wait_timeout(&pool->threads_all_idle, &pool->thcount_lock, timeout_ms);
344 if (ret == -1) {
345 perror("cond_wait_timeout");
346 }
347 break;
348 }
349
350 // Signal shutdown
351 atomic_store_explicit(&pool->shutdown, 1, memory_order_release);
352 lock_release(&pool->thcount_lock);
353
354 // Wake all threads
355 for (size_t i = 0; i < pool->num_threads; i++) {
356 cond_broadcast(&pool->threads[i]->queue.not_empty);
357 cond_broadcast(&pool->threads[i]->queue.not_full);
358 }
359
360 cond_broadcast(&pool->queue.not_empty);
361 cond_broadcast(&pool->queue.not_full);
362
363 // Wait for thread termination
364 while (atomic_load_explicit(&pool->num_threads_alive, memory_order_acquire) > 0) {
365 thread_yield();
366 }
367
368 // Cleanup
369 for (size_t i = 0; i < pool->num_threads; i++) {
370 thread* thp = pool->threads[i];
371 thread_join(thp->pthread, NULL);
372 ringbuffer_destroy(&thp->queue);
373 free(thp);
374 }
375
376 free(pool->threads);
377 ringbuffer_destroy(&pool->queue);
378 lock_free(&pool->thcount_lock);
379 cond_free(&pool->threads_all_idle);
380 free(pool);
381}
