106 }

175file_result_t filesize_tostring(uint64_t size, char* buf, size_t len) {
176 if (!buf || len < 8) { // Minimum for "1024.00 B"
177 return FILE_ERROR_INVALID_ARGS;
178 }
179
180 if (size == 0) {
181 int written = snprintf(buf, len, "0 B");
182 return (written > 0 && (size_t)written < len) ? FILE_SUCCESS : FILE_ERROR_INVALID_ARGS;
183 }
184
185 static const char* const units[] = {"B", "KB", "MB", "GB", "TB", "PB", "EB"};
186 static const size_t num_units = sizeof(units) / sizeof(units[0]);
187
188 size_t unit_index = 0;
189 double value = (double)size;
190
191 while (value >= 1024.0 && unit_index < num_units - 1) {
192 value /= 1024.0;
193 unit_index++;
194 }
195
196 double rounded = round(value);
197 int written = -1;
198
199 if (fabs(value - rounded) < HUMAN_SIZE_EPSILON) {
200 written = snprintf(buf, len, "%.0f %s", rounded, units[unit_index]);
201 } else {
202 written = snprintf(buf, len, "%.2f %s", value, units[unit_index]);
203 }
204
205 return (written > 0 && (size_t)written < len) ? FILE_SUCCESS : FILE_ERROR_INVALID_ARGS;
206}

274void* file_readall(const file_t* file, size_t* size_out) {
275 int64_t size = file_get_size(file);
276 if (!is_valid_file_size(size)) {
277 errno = (size < 0) ? EIO : EFBIG;
278 return NULL;
279 }
280
281 if (size == 0) {
282 if (size_out) *size_out = 0;
283 // Return a valid pointer for zero-sized files
284 return malloc(1);
285 }
286
287 void* buffer = malloc((size_t)size);
288 if (!buffer) {
289 errno = ENOMEM;
290 return NULL;
291 }
292
293 // Save current position and seek to beginning
294 int64_t orig_pos = file_tell(file);
295 if (fseek(file->stream, 0, SEEK_SET) != 0) {
296 free(buffer);
297 return NULL;
298 }
299
300 size_t bytes_read = file_read(file, buffer, 1, (size_t)size);
301
302 // Restore original position (best effort)
303 if (orig_pos >= 0) {
304 fseek(file->stream, (long)orig_pos, SEEK_SET);
305 }
306
307 if (bytes_read != (size_t)size) {
308 free(buffer);
309 errno = (ferror(file->stream)) ? EIO : EINVAL;
310 return NULL;
311 }
312
313 if (size_out) {
314 *size_out = bytes_read;
315 }
316
317 return buffer;
318}

367file_result_t file_copy(const file_t* src, file_t* dst) {
368 char buffer[COPY_BUFSIZE] = {0};
369 size_t bytes_read = 0;
370
371 // Clear any previous errors
372 clearerr(src->stream);
373 clearerr(dst->stream);
374
375 while ((bytes_read = file_read(src, buffer, 1, COPY_BUFSIZE)) > 0) {
376 if (file_write(dst, buffer, 1, bytes_read) != bytes_read) {
377 return FILE_ERROR_IO_FAILED;
378 }
379 }
380
381 // Check for read error
382 if (ferror(src->stream)) {
383 return FILE_ERROR_IO_FAILED;
384 }
385
386 // Flush destination
387 if (fflush(dst->stream) != 0) {
388 return FILE_ERROR_IO_FAILED;
389 }
390
391 return FILE_SUCCESS;
392}

394void* file_mmap(const file_t* file, size_t length, bool read_access, bool write_access) {
395 if (length == 0 || (!read_access && !write_access)) {
396 errno = EINVAL;
397 return NULL;
398 }
399
400#ifdef _WIN32
401 DWORD protect = write_access ? PAGE_READWRITE : PAGE_READONLY;
402 DWORD access = write_access ? FILE_MAP_WRITE : FILE_MAP_READ;
403
404 HANDLE mapping = CreateFileMapping(file->native_handle, NULL, protect, (DWORD)(length >> 32), (DWORD)length, NULL);
405 if (!mapping) {
406 errno = EIO;
407 return NULL;
408 }
409
410 void* addr = MapViewOfFile(mapping, access, 0, 0, length);
411 CloseHandle(mapping);
412
413 if (!addr) {
414 errno = EIO;
415 }
416 return addr;
417#else
418 int prot = 0;
419 if (read_access) prot |= PROT_READ;
420 if (write_access) prot |= PROT_WRITE;
421
422 void* addr = mmap(NULL, length, prot, MAP_SHARED, file->native_handle, 0);
423 return (addr == MAP_FAILED) ? NULL : addr;
424#endif
425}

459file_result_t file_seek(file_t* file, int64_t offset, int whence) {
460 // Flush any pending writes before seeking
461 if (fflush(file->stream) != 0) {
462 return FILE_ERROR_IO_FAILED;
463 }
464
465#ifdef _WIN32
466 DWORD move_method;
467 switch (whence) {
468 case SEEK_SET:
469 move_method = FILE_BEGIN;
470 break;
471 case SEEK_CUR:
472 move_method = FILE_CURRENT;
473 break;
474 case SEEK_END:
475 move_method = FILE_END;
476 break;
477 default:
478 errno = EINVAL;
479 return FILE_ERROR_INVALID_ARGS;
480 }
481
482 LARGE_INTEGER li = {.QuadPart = offset};
483 if (SetFilePointerEx(file->native_handle, li, NULL, move_method)) {
484 // Sync the FILE* stream position
485 fseek(file->stream, 0, SEEK_CUR);
486 return FILE_SUCCESS;
487 }
488 errno = EIO;
489 return FILE_ERROR_IO_FAILED;
490#else
491 if (lseek(file->native_handle, (off_t)offset, whence) == (off_t)-1) {
492 return FILE_ERROR_IO_FAILED; // errno already set
493 }
494
495 // Sync the FILE* stream position
496 if (fseek(file->stream, 0, SEEK_CUR) != 0) {
497 return FILE_ERROR_IO_FAILED;
498 }
499
500 return FILE_SUCCESS;
501#endif
502}
