366static ARENA_NONNULL_ ARENA_INLINE ARENA_ATTR_MALLOC
367ARENA_ATTR_ALLOC_SIZE(2) void* arena_alloc_align(Arena* arena, size_t size, size_t alignment) {
368#ifdef ARENA_ABORT_ON_OOM
369 /* Preserve the returns_nonnull contract: treat size==0 as size==1. */
370 if (size == 0) size = 1;
371#else
372 if (size == 0) return NULL;
373#endif
374
375 uintptr_t curr = (uintptr_t)arena->curr;
376 uintptr_t aligned = (curr + alignment - 1) & ~(uintptr_t)(alignment - 1);
377 uintptr_t next = aligned + size;
378
379 if (ARENA_LIKELY(next <= (uintptr_t)arena->end)) {
380 arena->curr = (char*)next;
381 /* Prefetch the allocated region; hides DRAM latency when the caller
382 * immediately writes the object (the common case). */
383 ARENA_PREFETCH((void*)aligned);
384 return (void*)aligned;
385 }
386
387 void* ptr = _arena_alloc_slow(arena, size, alignment);
388 if (ARENA_UNLIKELY(!ptr)) {
389#ifdef ARENA_ABORT_ON_OOM
390 ARENA_OOM_HANDLER(size);
391 ARENA_UNREACHABLE();
392#else
393 return NULL;
394#endif
395 }
396 return ptr;
397}

425static ARENA_NONNULL_ ARENA_INLINE ARENA_ATTR_MALLOC ARENA_ATTR_ALLOC_SIZE(2) void* arena_alloc_unaligned(Arena* arena,
426 size_t size) {

428 if (size == 0) size = 1;
429#else
430 if (size == 0) return NULL;
431#endif

433 char* ptr = arena->curr;
434 char* next = ptr + size;

437 arena->curr = next;
438 ARENA_PREFETCH(ptr);

481static ARENA_INLINE void* arena_realloc(Arena* arena, void* old_ptr, size_t old_size, size_t new_size,
482 size_t alignment) {
483 if (!old_ptr || old_size == 0) { return arena_alloc_align(arena, new_size, alignment); }
484
485 if (new_size == 0) {
486 /* Shrink to nothing: reclaim bytes only if this was the last alloc. */
487 if ((char*)old_ptr + old_size == arena->curr) { arena->curr = (char*)old_ptr; }
488 /* NULL here is a defined contract value, not an OOM — do not invoke
489 * ARENA_OOM_HANDLER. */
490 return NULL;
491 }
492
493 /* Fast path: old_ptr was the last allocation in the current block. */
494 if ((char*)old_ptr + old_size == arena->curr) {
495 char* new_end = (char*)old_ptr + new_size;
496 if (new_size <= old_size || new_end <= arena->end) {
497 arena->curr = new_end;
498 return old_ptr;
499 }
500 }
501
502 /* Slow path: fresh allocation + copy; old bytes are stranded until reset. */
503 void* new_ptr = arena_alloc_align(arena, new_size, alignment);
504 if (ARENA_UNLIKELY(!new_ptr)) {
505 /* arena_alloc_align already invoked ARENA_OOM_HANDLER if the macro is
506 * set, so we only reach here in the non-abort build. */
507 return NULL;
508 }
509 memcpy(new_ptr, old_ptr, old_size < new_size ? old_size : new_size);
510 return new_ptr;
511}

589static ARENA_INLINE bool arena_alloc_batch(Arena* arena, const size_t* sizes, size_t count, void** out_ptrs) {
590 if (!arena || !sizes || !out_ptrs || count == 0) return false;
591
592 const size_t mask = ARENA_DEFAULT_ALIGN - 1;
593 size_t total = 0;
594 for (size_t i = 0; i < count; ++i) {
595 size_t rounded = (total + mask) & ~mask;
596 /* Detect integer overflow in the size accumulation. This is a
597 * caller bug, not OOM — return false without invoking the OOM handler
598 * regardless of ARENA_ABORT_ON_OOM. */
599 if (ARENA_UNLIKELY(sizes[i] > (size_t)-1 - rounded)) return false;
600 total = rounded + sizes[i];
601 }
602
603 char* base = (char*)arena_alloc(arena, total);
604 if (ARENA_UNLIKELY(!base)) {
605 /* arena_alloc already called ARENA_OOM_HANDLER if the macro is set. */
606 return false;
607 }
608
609 char* cur = base;
610 for (size_t i = 0; i < count; ++i) {
611 cur = (char*)(((uintptr_t)cur + mask) & ~mask);
612 out_ptrs[i] = cur;
613 cur += sizes[i];
614 }
615 return true;
616}

634static ARENA_INLINE char* arena_strdup(Arena* arena, const char* str) {
635 /* NULL arguments are a programming error, not OOM. Return NULL silently
636 * (or rely on the caller to guard); do not invoke ARENA_OOM_HANDLER. */
637 if (ARENA_UNLIKELY(!arena || !str)) return NULL;
638 size_t len = strlen(str);
639 char* dup = (char*)arena_alloc_unaligned(arena, len + 1);
640 if (ARENA_UNLIKELY(!dup)) return NULL;
641 memcpy(dup, str, len + 1);
642 return dup;
643}
