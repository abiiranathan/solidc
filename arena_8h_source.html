246static ARENA_INLINE void* arena_alloc_align(Arena* arena, size_t size, size_t alignment) {
247 // Calculate aligned address
248 uintptr_t curr = (uintptr_t)arena->curr;
249 uintptr_t mask = alignment - 1;
250 uintptr_t aligned_addr = (curr + mask) & ~mask;
251 uintptr_t new_curr = aligned_addr + size;
252
253 // Fast path: allocation fits in committed memory
254 if (ARENA_LIKELY(new_curr <= (uintptr_t)arena->end)) {
255 arena->curr = (char*)new_curr;
256 return (void*)aligned_addr;
257 }
258
259 // Slow path: need to commit more pages
260 return _arena_alloc_slow(arena, size, alignment);
261}

371static ARENA_INLINE bool arena_alloc_batch(Arena* arena, const size_t sizes[], size_t count, void* out_ptrs[]) {
372 if (arena == NULL || sizes == NULL || out_ptrs == NULL || count == 0) {
373 return false;
374 }
375
376 // Calculate total size needed with alignment padding for each block
377 size_t total_size = 0;
378 const size_t alignment = ARENA_DEFAULT_ALIGN;
379 const size_t align_mask = alignment - 1;
380
381 for (size_t i = 0; i < count; ++i) {
382 // Each block needs to be aligned
383 total_size = (total_size + align_mask) & ~align_mask;
384 total_size += sizes[i];
385 }
386
387 // Allocate the entire batch as one contiguous block
388 char* base = (char*)arena_alloc(arena, total_size);
389 if (base == NULL) {
390 return false;
391 }
392
393 // Divide the allocated block among the requested pointers
394 char* current = base;
395 for (size_t i = 0; i < count; ++i) {
396 // Align current pointer
397 uintptr_t addr = (uintptr_t)current;
398 addr = (addr + align_mask) & ~align_mask;
399 current = (char*)addr;
400
401 out_ptrs[i] = current;
402 current += sizes[i];
403 }
404
405 return true;
406}

428static ARENA_INLINE char* arena_strdup(Arena* arena, const char* str) {
429 if (arena == NULL || str == NULL) {
430 return NULL;
431 }
432
433 const size_t len = strlen(str);
434 char* dup = (char*)arena_alloc(arena, len + 1);
435 if (dup == NULL) {
436 return NULL;
437 }
438
439 memcpy(dup, str, len);
440 dup[len] = '\0';
441 return dup;
442}
